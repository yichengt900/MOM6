220 subroutine convert_iob_to_fluxes(IOB, fluxes, index_bounds, Time, G, CS, state, restore_salt, restore_temp)
221  type(ice_ocean_boundary_type), intent(in), target :: iob
222  type(forcing), intent(inout) :: fluxes
223  integer, dimension(4), intent(in) :: index_bounds

230 ! This subroutine translates the Ice_ocean_boundary_type into a
231 ! MOM forcing type, including changes of units, sign conventions,
232 ! and puting the fields into arrays with MOM-standard halos.
233 
234 ! Arguments:
235 ! IOB ice-ocean boundary type w/ fluxes to drive ocean in a coupled model
236 ! (out) fluxes - A structure containing pointers to any possible
237 ! forcing fields. Unused fields have NULL ptrs.
238 ! (in) index_bounds - the i- and j- size of the arrays in IOB.
239 ! (in) Time - The time of the fluxes, used for interpolating the salinity
240 ! to the right time, when it is being restored.
241 ! (in) G - The ocean's grid structure.
242 ! (in) CS - A pointer to the control structure returned by a previous
243 ! call to surface_forcing_init.
244 ! (in) state - A structure containing fields that describe the
245 ! surface state of the ocean.
246 ! (in) restore_salt - if true, salinity is restored to a target value.
247 ! (in) restore_temp - if true, temperature is restored to a target value.
248 
249  real, dimension(SZIB_(G),SZJB_(G)) :: &
250  taux_at_q, & ! Zonal wind stresses at q points (Pa)
251  tauy_at_q ! Meridional wind stresses at q points (Pa)
252 
253  real, dimension(SZI_(G),SZJ_(G)) :: &
254  taux_at_h, & ! Zonal wind stresses at h points (Pa)
255  tauy_at_h, & ! Meridional wind stresses at h points (Pa)
256  data_restore, & ! The surface value toward which to restore (g/kg or degC)
257  sst_anom, & ! Instantaneous sea surface temperature anomalies from a target value (deg C)
258  sss_anom, & ! Instantaneous sea surface salinity anomalies from a target value (g/kg)
259  sss_mean, & ! A (mean?) salinity about which to normalize local salinity
260  ! anomalies when calculating restorative precipitation anomalies (g/kg)
261  pme_adj, & ! The adjustment to PminusE that will cause the salinity
262  ! to be restored toward its target value (kg/(m^2 * s))
263  net_fw, & ! The area integrated net freshwater flux into the ocean (kg/s)
264  net_fw2, & ! The area integrated net freshwater flux into the ocean (kg/s)
265  work_sum, & ! A 2-d array that is used as the work space for a global
266  ! sum, used with units of m2 or (kg/s)
267  open_ocn_mask ! a binary field indicating where ice is present based on frazil criteria
268 
269  real :: gustiness ! unresolved gustiness that contributes to ustar (Pa)
270  real :: irho0 ! inverse of the mean density in (m^3/kg)
271  real :: taux2, tauy2 ! squared wind stresses (Pa^2)
272  real :: tau_mag ! magnitude of the wind stress (Pa)
273  real :: i_gearth ! 1.0 / G%G_Earth (s^2/m)
274  real :: kv_rho_ice ! (CS%kv_sea_ice / CS%density_sea_ice) ( m^5/(s*kg) )
275  real :: mass_ice ! mass of sea ice at a face (kg/m^2)
276  real :: mass_eff ! effective mass of sea ice for rigidity (kg/m^2)
277 
278  integer :: wind_stagger ! AGRID, BGRID_NE, or CGRID_NE (integers from MOM_domains)
279  integer :: i, j, is, ie, js, je, isq, ieq, jsq, jeq, i0, j0
280  integer :: isd, ied, jsd, jed, isdb, iedb, jsdb, jedb, isr, ier, jsr, jer
281  integer :: isc_bnd, iec_bnd, jsc_bnd, jec_bnd
282 
283  logical :: restore_salinity ! local copy of the argument restore_salt, if it
284  ! is present, or false (no restoring) otherwise.
285  logical :: restore_sst ! local copy of the argument restore_temp, if it
286  ! is present, or false (no restoring) otherwise.
287  real :: delta_sss ! temporary storage for sss diff from restoring value
288  real :: delta_sst ! temporary storage for sst diff from restoring value
289 
290  real :: c_p ! heat capacity of seawater ( J/(K kg) )

292  call cpu_clock_begin(id_clock_forcing)
293 
294  isc_bnd = index_bounds(1) ; iec_bnd = index_bounds(2)
295  jsc_bnd = index_bounds(3) ; jec_bnd = index_bounds(4)
296  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec
297  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
298  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
299  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
300  isr = is-isd+1 ; ier = ie-isd+1 ; jsr = js-jsd+1 ; jer = je-jsd+1
301 
302  c_p = fluxes%C_p
303  irho0 = 1.0/cs%Rho0
304  open_ocn_mask(:,:) = 1.0
305  pme_adj(:,:) = 0.0
306  fluxes%vPrecGlobalAdj = 0.0
307  fluxes%vPrecGlobalScl = 0.0
308  fluxes%saltFluxGlobalAdj = 0.0
309  fluxes%saltFluxGlobalScl = 0.0
310  fluxes%netFWGlobalAdj = 0.0
311  fluxes%netFWGlobalScl = 0.0
312 
313  restore_salinity = .false.
314  if (present(restore_salt)) restore_salinity = restore_salt
315  restore_sst = .false.
316  if (present(restore_temp)) restore_sst = restore_temp
317 
318  ! allocation and initialization if this is the first time that this
319  ! flux type has been used.
320  if (fluxes%dt_buoy_accum < 0) then
321  call allocate_forcing_type(g, fluxes, stress=.true., ustar=.true., &

326  call safe_alloc_ptr(fluxes%sw_nir_dir,isd,ied,jsd,jed)

331 
332  call safe_alloc_ptr(fluxes%salt_flux,isd,ied,jsd,jed)

337  call safe_alloc_ptr(fluxes%ustar_tidal,isd,ied,jsd,jed)

340  call safe_alloc_ptr(fluxes%heat_added,isd,ied,jsd,jed)

342  endif

344  do j=js-2,je+2 ; do i=is-2,ie+2
345  fluxes%TKE_tidal(i,j) = cs%TKE_tidal(i,j)
346  fluxes%ustar_tidal(i,j) = cs%ustar_tidal(i,j)

350  call safe_alloc_ptr(fluxes%rigidity_ice_u,isdb,iedb,jsd,jed)

353 
354  if (restore_temp) call safe_alloc_ptr(fluxes%heat_added,isd,ied,jsd,jed)

356  fluxes%dt_buoy_accum = 0.0
357  endif ! endif for allocation and initialization
358

360  fluxes%heat_added(:,:)=0.0
361  fluxes%salt_flux_added(:,:)=0.0
362  endif
363

365  if (cs%area_surf < 0.0) then
366  do j=js,je ; do i=is,ie
367  work_sum(i,j) = g%areaT(i,j) * g%mask2dT(i,j)
368  enddo ; enddo
369  cs%area_surf = reproducing_sum(work_sum, isr, ier, jsr, jer)
370  endif ! endif for allocation and initialization
371 
372  do j=js,je ; do i=is,ie
373  fluxes%salt_flux(i,j) = 0.0
374  fluxes%vprec(i,j) = 0.0
375  enddo ; enddo
376 
377  ! Salinity restoring logic
378  if (restore_salinity) then
379  call time_interp_external(cs%id_srestore,time,data_restore)
380  ! open_ocn_mask indicates where to restore salinity (1 means restore, 0 does not)
381  open_ocn_mask(:,:) = 1.0
382  if (cs%mask_srestore_under_ice) then ! Do not restore under sea-ice
383  do j=js,je ; do i=is,ie
384  if (state%SST(i,j) .le. -0.0539*state%SSS(i,j)) open_ocn_mask(i,j)=0.0
385  enddo; enddo
386  endif
387  if (cs%salt_restore_as_sflux) then
388  do j=js,je ; do i=is,ie
389  delta_sss = data_restore(i,j)- state%SSS(i,j)
390  delta_sss = sign(1.0,delta_sss)*min(abs(delta_sss),cs%max_delta_srestore)
391  fluxes%salt_flux(i,j) = 1.e-3*g%mask2dT(i,j) * (cs%Rho0*cs%Flux_const)* &
392  (cs%basin_mask(i,j)*open_ocn_mask(i,j)) *delta_sss ! kg Salt m-2 s-1
393  enddo; enddo
394  if (cs%adjust_net_srestore_to_zero) then
395  if (cs%adjust_net_srestore_by_scaling) then
396  call adjust_area_mean_to_zero(fluxes%salt_flux, g, fluxes%saltFluxGlobalScl)
397  fluxes%saltFluxGlobalAdj = 0.
398  else
399  work_sum(is:ie,js:je) = g%areaT(is:ie,js:je)*fluxes%salt_flux(is:ie,js:je)
400  fluxes%saltFluxGlobalAdj = reproducing_sum(work_sum(:,:), isr,ier, jsr,jer)/cs%area_surf
401  fluxes%salt_flux(is:ie,js:je) = fluxes%salt_flux(is:ie,js:je) - fluxes%saltFluxGlobalAdj
402  endif
403  endif
404  fluxes%salt_flux_added(is:ie,js:je) = fluxes%salt_flux(is:ie,js:je) ! Diagnostic
405  else
406  do j=js,je ; do i=is,ie
407  if (g%mask2dT(i,j) > 0.5) then
408  delta_sss = state%SSS(i,j) - data_restore(i,j)
409  delta_sss = sign(1.0,delta_sss)*min(abs(delta_sss),cs%max_delta_srestore)
410  fluxes%vprec(i,j) = (cs%basin_mask(i,j)*open_ocn_mask(i,j))* &
411  (cs%Rho0*cs%Flux_const) * &
412  delta_sss / (0.5*(state%SSS(i,j) + data_restore(i,j)))
413  endif
414  enddo; enddo
415  if (cs%adjust_net_srestore_to_zero) then
416  if (cs%adjust_net_srestore_by_scaling) then
417  call adjust_area_mean_to_zero(fluxes%vprec, g, fluxes%vPrecGlobalScl)
418  fluxes%vPrecGlobalAdj = 0.
419  else
420  work_sum(is:ie,js:je) = g%areaT(is:ie,js:je)*fluxes%vprec(is:ie,js:je)
421  fluxes%vPrecGlobalAdj = reproducing_sum(work_sum(:,:), isr, ier, jsr, jer) / cs%area_surf
422  do j=js,je ; do i=is,ie
423  fluxes%vprec(i,j) = ( fluxes%vprec(i,j) - fluxes%vPrecGlobalAdj ) * g%mask2dT(i,j)
424  enddo; enddo
425  endif
426  endif
427  endif
428  endif

740  endif

748 !> Adds flux adjustments obtained via data_override

792

810  if (overrode_x .or. overrode_y) then
811  if (.not. (overrode_x .and. overrode_y)) call mom_error(fatal,"apply_flux_adjustments: "//&
812  "Both taux_adj and tauy_adj must be specified, or neither, in data_table")
813 
814  ! Rotate winds
815  call pass_vector(tempx_at_h, tempy_at_h, g%Domain, to_all, agrid)
816  do j=g%jsc-1,g%jec+1 ; do i=g%isc-1,g%iec+1
817  dlondx = g%geoLonCu(i,j) - g%geoLonCu(i-1,j)
818  dlondy = g%geoLonCv(i,j) - g%geoLonCv(i,j-1)
819  rdlon = sqrt( dlondx * dlondx + dlondy * dlondy )
820  if (rdlon > 0.) rdlon = 1. / rdlon
821  cosa = dlondx * rdlon
822  sina = dlondy * rdlon
823  zonal_tau = tempx_at_h(i,j)
824  merid_tau = tempy_at_h(i,j)
825  tempx_at_h(i,j) = cosa * zonal_tau - sina * merid_tau
826  tempy_at_h(i,j) = sina * zonal_tau + cosa * merid_tau
827  enddo ; enddo
828 
829  ! Average to C-grid locations
830  do j=g%jsc,g%jec ; do i=g%isc-1,g%iec
831  fluxes%taux(i,j) = fluxes%taux(i,j) + 0.5 * ( tempx_at_h(i,j) + tempx_at_h(i+1,j) )
832  enddo ; enddo
833 
834  do j=g%jsc-1,g%jec ; do i=g%isc,g%iec
835  fluxes%tauy(i,j) = fluxes%tauy(i,j) + 0.5 * ( tempy_at_h(i,j) + tempy_at_h(i,j+1) )
836  enddo ; enddo
837  endif ! overrode_x .or. overrode_y
838 
839 end subroutine apply_flux_adjustments
