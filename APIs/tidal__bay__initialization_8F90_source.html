38 !> This subroutine sets the properties of flow at open boundary conditions.
39 subroutine tidal_bay_set_obc_data(OBC, G, h, Time)
40  type(ocean_obc_type), pointer :: obc !< This open boundary condition type specifies
41  !! whether, where, and what open boundary
42  !! conditions are used.
43  type(ocean_grid_type), intent(in) :: g !< The ocean's grid structure.
44  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(in) :: h !< layer thickness.
45  type(time_type), intent(in) :: time !< model time.
46 
47  ! The following variables are used to set up the transport in the tidal_bay example.
48  real :: time_sec, cff, cff2, tide_flow
49  real :: my_area, my_flux
50  real :: pi
51  character(len=40) :: mod = "tidal_bay_set_OBC_data" ! This subroutine's name.
52  integer :: i, j, k, itt, is, ie, js, je, isd, ied, jsd, jed, nz
53  integer :: isdb, iedb, jsdb, jedb
54 
55  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
56  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
57  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
58 
59  pi = 4.0*atan(1.0) ;
60 
61  if (.not.associated(obc)) return
62 
63  time_sec = time_type_to_real(time)
64  cff = 0.1*sin(2.0*pi*time_sec/(12.0*3600.0))
65  tide_flow = 3.0e6
66  my_area=0.0
67  my_flux=0.0
68  do j=jsd,jed ; do i=isdb,iedb
69  if (obc%OBC_segment_u(i,j) /= obc_none) then
70  do k=1,nz
71  cff2 = h(i,j,k)*g%dyCu(i,j)
72  my_area = my_area + cff2
73  enddo
74  endif
75  enddo ; enddo
76  my_flux = -tide_flow*sin(2.0*pi*time_sec/(12.0*3600.0))
77 
78  do j=jsd,jed ; do i=isdb,iedb
79  if (obc%OBC_segment_u(i,j) /= obc_none) then
80  obc%eta_outer_u(i,j) = cff
81  obc%ubt_outer(i,j) = my_flux/my_area
82  endif
83  enddo ; enddo
84  do j=jsdb,jedb ; do i=isd,ied
85  if (obc%OBC_segment_v(i,j) /= obc_none) then
86  obc%eta_outer_v(i,j) = cff
87  obc%vbt_outer(i,j) = 0.0
88  endif
89  enddo ; enddo
90 
91 end subroutine tidal_bay_set_obc_data
