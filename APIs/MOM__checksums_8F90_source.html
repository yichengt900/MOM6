78 !> chksum_h_2d performs checksums on a 2d array staggered at tracer points.
79 subroutine chksum_h_2d(array, mesg, HI, haloshift)
80  type(hor_index_type), intent(in) :: HI !< A horizontal index type
81  real, dimension(HI%isd:,HI%jsd:), intent(in) :: array !< The array to be checksummed
82  character(len=*), intent(in) :: mesg !< An identifying message
83  integer, optional, intent(in) :: haloshift !< The width of halos to check (default 0)
84 
85  integer :: bc0,bcSW,bcSE,bcNW,bcNE,hshift
86 
87  if (checkfornans) then
88  if (is_nan(array(hi%isc:hi%iec,hi%jsc:hi%jec))) &
89  call chksum_error(fatal, 'NaN detected: '//trim(mesg))
90 ! if (is_NaN(array)) &
91 ! call chksum_error(FATAL, 'NaN detected in halo: '//trim(mesg))
92  endif
93 
94  if (calculatestatistics) call substats(hi, array, mesg)
95 
96  if (.not.writechksums) return
97 
98  hshift=default_shift
99  if (present(haloshift)) hshift=haloshift
100  if (hshift<0) hshift=hi%ied-hi%iec
101 
102  if ( hi%isc-hshift<hi%isd .or. hi%iec+hshift>hi%ied .or. &
103  hi%jsc-hshift<hi%jsd .or. hi%jec+hshift>hi%jed ) then
104  write(0,*) 'chksum_h_2d: haloshift =',hshift
105  write(0,*) 'chksum_h_2d: isd,isc,iec,ied=',hi%isd,hi%isc,hi%iec,hi%ied
106  write(0,*) 'chksum_h_2d: jsd,jsc,jec,jed=',hi%jsd,hi%jsc,hi%jec,hi%jed
107  call chksum_error(fatal,'Error in chksum_h_2d '//trim(mesg))
108  endif
109 
110  bc0=subchk(array, hi, 0, 0)
111 
112  if (hshift==0) then
113  if (is_root_pe()) call chk_sum_msg("h-point:",bc0,mesg)
114  return
115  endif
116 
117  bcsw=subchk(array, hi, -hshift, -hshift)
118  bcse=subchk(array, hi, hshift, -hshift)
119  bcnw=subchk(array, hi, -hshift, hshift)
120  bcne=subchk(array, hi, hshift, hshift)
121 
122  if (is_root_pe()) call chk_sum_msg("h-point:",bc0,bcsw,bcse,bcnw,bcne,mesg)

126  integer function subchk(array, HI, di, dj)
127  type(hor_index_type), intent(in) :: hi
128  real, dimension(HI%isd:,HI%jsd:), intent(in) :: array
129  integer, intent(in) :: di, dj
130  integer :: bitcount, i, j, bc
131  subchk = 0
132  do j=hi%jsc+dj,hi%jec+dj; do i=hi%isc+di,hi%iec+di
133  bc = bitcount(abs(array(i,j)))
134  subchk = subchk + bc
135  enddo; enddo
136  call sum_across_pes(subchk)
137  subchk=mod(subchk,1000000000)
138  end function subchk

140  subroutine substats(HI, array, mesg)
141  type(hor_index_type), intent(in) :: HI
142  real, dimension(HI%isd:,HI%jsd:), intent(in) :: array
143  character(len=*), intent(in) :: mesg
144  integer :: i, j, n
145  real :: aMean, aMin, aMax
146 
147  amin = array(hi%isc,hi%jsc)
148  amax = array(hi%isc,hi%jsc)
149  n = 0
150  do j=hi%jsc,hi%jec ; do i=hi%isc,hi%iec
151  amin = min(amin, array(i,j))
152  amax = max(amax, array(i,j))
153  n = n + 1
154  enddo ; enddo
155  amean = reproducing_sum(array(hi%isc:hi%iec,hi%jsc:hi%jec))
156  call sum_across_pes(n)
157  call min_across_pes(amin)
158  call max_across_pes(amax)
159  amean = amean / real(n)
160  if (is_root_pe()) call chk_sum_msg("h-point:",amean,amin,amax,mesg)
161  end subroutine substats

163 end subroutine chksum_h_2d

167 !> chksum_B_2d performs checksums on a 2d array staggered at corner points.
168 subroutine chksum_b_2d(array, mesg, HI, haloshift, symmetric)
169  type(hor_index_type), intent(in) :: HI !< A horizontal index type
170  real, dimension(HI%IsdB:,HI%JsdB:), &
171  intent(in) :: array !< The array to be checksummed
172  character(len=*), intent(in) :: mesg !< An identifying message
173  integer, optional, intent(in) :: haloshift !< The width of halos to check (default 0)
174  logical, optional, intent(in) :: symmetric !< If true, do the checksums on the
175  !! full symmetric computational domain.
176 
177  integer :: bc0,bcSW,bcSE,bcNW,bcNE,hshift
178  logical :: sym
179 
180  if (checkfornans) then
181  if (is_nan(array(hi%IscB:hi%IecB,hi%JscB:hi%JecB))) &
182  call chksum_error(fatal, 'NaN detected: '//trim(mesg))
183 ! if (is_NaN(array)) &
184 ! call chksum_error(FATAL, 'NaN detected in halo: '//trim(mesg))
185  endif
186 
187  if (calculatestatistics) call substats(hi, array, mesg)
188 
189  if (.not.writechksums) return
190 
191  hshift=default_shift
192  if (present(haloshift)) hshift=haloshift
193  if (hshift<0) hshift=hi%ied-hi%iec
194 
195  if ( hi%isc-hshift<hi%isd .or. hi%iec+hshift>hi%ied .or. &
196  hi%jsc-hshift<hi%jsd .or. hi%jec+hshift>hi%jed ) then
197  write(0,*) 'chksum_B_2d: haloshift =',hshift
198  write(0,*) 'chksum_B_2d: isd,isc,iec,ied=',hi%isd,hi%isc,hi%iec,hi%ied
199  write(0,*) 'chksum_B_2d: jsd,jsc,jec,jed=',hi%jsd,hi%jsc,hi%jec,hi%jed
200  call chksum_error(fatal,'Error in chksum_B_2d '//trim(mesg))
201  endif
202 
203  sym = .false. ; if (present(symmetric)) sym = symmetric
204 
205  bc0=subchk(array, hi, 0, 0)
206 
207  if ((hshift==0) .and. .not.sym) then
208  if (is_root_pe()) call chk_sum_msg("B-point:",bc0,mesg)
209  return
210  endif
211 
212  if (sym) then
213  bcsw=subchk(array, hi, -hshift-1, -hshift-1)
214  bcse=subchk(array, hi, hshift, -hshift-1)
215  bcnw=subchk(array, hi, -hshift-1, hshift)
216  else
217  bcsw=subchk(array, hi, -hshift, -hshift)
218  bcse=subchk(array, hi, hshift, -hshift)
219  bcnw=subchk(array, hi, -hshift, hshift)
220  endif
221  bcne=subchk(array, hi, hshift, hshift)
222 
223  if (is_root_pe()) call chk_sum_msg("B-point:",bc0,bcsw,bcse,bcnw,bcne,mesg)
