38 !> Initialization of thicknesses in 2D Rossby front test
39 subroutine rossby_front_initialize_thickness(h, G, GV, param_file )
40  type(ocean_grid_type), intent(in) :: g !< Grid structure
41  type(verticalgrid_type), intent(in) :: gv !< Vertical grid structure
42  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(out) :: h !< Thickness
43  type(param_file_type), intent(in) :: param_file !< Parameter file handle
44 
45  integer :: i, j, k, is, ie, js, je, nz
46  real :: tz, dml, eta, stretch, h0
47  real :: min_thickness, t_range, drho_dt
48  character(len=40) :: verticalcoordinate
49 
50  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
51 
52  call mom_mesg("Rossby_front_2d_initialization.F90, Rossby_front_initialize_thickness: setting thickness")
53 
54  ! Read parameters needed to set thickness
55  call get_param(param_file, mod, "MIN_THICKNESS", min_thickness, &
56  'Minimum layer thickness',units='m',default=1.e-3)
57  call get_param(param_file, mod, "REGRIDDING_COORDINATE_MODE", verticalcoordinate, &
58  default=default_coordinate_mode)
59  call get_param(param_file, mod, "T_RANGE", t_range, 'Initial temperature range', units='C', default=0.0)
60  call get_param(param_file, mod, "DRHO_DT", drho_dt, default=-0.2, do_not_log=.true.)
61 
62  tz = t_range / g%max_depth
63 
64  select case ( coordinatemode(verticalcoordinate) )
65 
66  case (regridding_layer, regridding_rho)
67  do j = g%jsc,g%jec ; do i = g%isc,g%iec
68  dml = hml( g, g%geoLatT(i,j) )
69  eta = -( -drho_dt / gv%Rho0 ) * tz * 0.5 * ( dml * dml )
70  stretch = ( ( g%max_depth + eta ) / g%max_depth )
71  h0 = ( g%max_depth / real(nz) ) * stretch
72  do k = 1, nz
73  h(i,j,k) = h0
74  enddo
75  end do ; end do
76 
77  case (regridding_zstar, regridding_sigma)
78  do j = g%jsc,g%jec ; do i = g%isc,g%iec
79  dml = hml( g, g%geoLatT(i,j) )
80  eta = -( -drho_dt / gv%Rho0 ) * tz * 0.5 * ( dml * dml )
81  stretch = ( ( g%max_depth + eta ) / g%max_depth )
82  h0 = ( g%max_depth / real(nz) ) * stretch
83  do k = 1, nz
84  h(i,j,k) = h0
85  enddo
86  end do ; end do
87 
88  case default
89  call mom_error(fatal,"Rossby_front_initialize: "// &
90  "Unrecognized i.c. setup - set REGRIDDING_COORDINATE_MODE")

92  end select

139 subroutine rossby_front_initialize_velocity(u, v, h, G, GV, param_file)
140  type(ocean_grid_type), intent(in) :: g !< Grid structure
141  type(verticalgrid_type), intent(in) :: gv !< Vertical grid structure
142  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(out) :: u !< i-component of velocity [m/s]
143  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(out) :: v !< j-component of velocity [m/s]
144  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(in) :: h !< Thickness [H]

151  real :: dudt ! Factor to convert dT/dy into dU/dz, g*alpha/f
152  real :: drho_dt, zi, zc, zm, f, ty, dml, hatu
153  integer :: i, j, k, is, ie, js, je, nz
154  character(len=40) :: verticalcoordinate
155 
156  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
157 
158  call get_param(param_file, mod, "REGRIDDING_COORDINATE_MODE", verticalcoordinate, &
159  default=default_coordinate_mode)
160  call get_param(param_file, mod, "T_RANGE", t_range, 'Initial temperature range', units='C', default=0.0)
161  call get_param(param_file, mod, "DRHO_DT", drho_dt, default=-0.2, do_not_log=.true.)
162 
163  v(:,:,:) = 0.0
164  u(:,:,:) = 0.0
165 
166  do j = g%jsc,g%jec ; do i = g%isc-1,g%iec+1
167  f = 0.5*( g%CoriolisBu(i,j) + g%CoriolisBu(i,j-1) )
168  dudt = 0.0 ; if (abs(f) > 0.0) &
169  dudt = ( gv%g_Earth * drho_dt ) / ( f * gv%Rho0 )
170  dml = hml( g, g%geoLatT(i,j) )
171  ty = dtdy( g, t_range, g%geoLatT(i,j) )
172  zi = 0.
173  do k = 1, nz
174  hatu = 0.5*(h(i,j,k)+h(i+1,j,k))
175  zi = zi - hatu ! Bottom interface position
176  zc = zi - 0.5*hatu ! Position of middle of cell
177  zm = max( zc + dml, 0. ) ! Height above bottom of mixed layer
178  u(i,j,k) = dudt * ty * zm ! Thermal wind starting at base of ML
179  enddo
180  enddo ; enddo
181 
182 end subroutine rossby_front_initialize_velocity

184 !> Pseudo coordinate across domain used by Hml() and dTdy()
185 !! returns a coordinate from -PI/2 .. PI/2 squashed towards the
186 !! center of the domain.
187 real function ypseudo( G, lat )
188  type(ocean_grid_type), intent(in) :: g !< Grid structure
189  real, intent(in) :: lat !< Latitude
190  ! Local

199 !> Analytic prescription of mixed layer depth in 2d Rossby front test
200 real function hml( G, lat )
201  type(ocean_grid_type), intent(in) :: g !< Grid structure
202  real, intent(in) :: lat !< Latitude
203  ! Local
204  real :: dhml, hmlmean

212 !> Analytic prescription of mixed layer temperature gradient in 2d Rossby front test
213 real function dtdy( G, dT, lat )
214  type(ocean_grid_type), intent(in) :: g !< Grid structure
215  real, intent(in) :: dt !< Top to bottom temperature difference
216  real, intent(in) :: lat !< Latitude
217  ! Local
218  real :: pi, dhml, dhdy
219  real :: km = 1.e3 ! AXIS_UNITS = 'k' (1000 m)
220 
221  pi = 4.0 * atan(1.0)
222  dhml = 0.5 * ( hmlmax - hmlmin ) * g%max_depth
223  dhdy = dhml * ( pi / ( frontfractionalwidth * g%len_lat * km ) ) * cos( ypseudo(g, lat) )
224  dtdy = -( dt / g%max_depth ) * dhdy
225

229 !> \namespace Rossby_front_2d_initialization
230 !!
231 !! \section section_Rossby_front_2d Description of the 2d Rossby front initial conditions
232 !!
233 !! Consistent with a linear equation of state, the system has a constant stratification
234 !! below the mixed layer, stratified in temperature only. Isotherms are flat below the
235 !! mixed layer and vertical within. Salinity is constant. The mixed layer has a half sine
236 !! form so that there are no mixed layer or temperature gradients at the side walls.
237 !!
238 !! Below the mixed layer the potential temperature, \f$\theta(z)\f$, is given by
239 !! \f[ \theta(z) = \theta_0 - \Delta \theta \left( z + h_{ML} \right) \f]
240 !! where \f$ \theta_0 \f$ and \f$ \Delta \theta \f$ are external model parameters.
241 !!
242 !! The depth of the mixed layer, \f$H_{ML}\f$ is
243 !! \f[ h_{ML}(y) = h_{min} + \left( h_{max} - h_{min} \right) \cos{\pi y/L} \f].
244 !! The temperature in mixed layer is given by the reference temperature at \f$z=h_{ML}\f$
245 !! so that
246 !! \f[ \theta(y,z) =
247 !! \theta_0 - \Delta \theta \left( z + h_{ML} \right) & \forall & z < h_{ML}(y) T.B.D.
248 !! \f]
