59 !> Parse a string parameter specifying the coordinate mode and
60 !! return the appropriate enumerated integer
61 function coordinatemode(string)
62  integer :: coordinatemode !< Enumerated integer indicating coordinate mode
63  character(len=*), intent(in) :: string !< String to indicate coordinate mode
64  select case ( uppercase(trim(string)) )
65  case (trim(regridding_layer_string)); coordinatemode = regridding_layer
66  case (trim(regridding_zstar_string)); coordinatemode = regridding_zstar
67  case (trim(regridding_zstar_string_old)); coordinatemode = regridding_zstar
68  case (trim(regridding_rho_string)); coordinatemode = regridding_rho
69  case (trim(regridding_sigma_string)); coordinatemode = regridding_sigma
