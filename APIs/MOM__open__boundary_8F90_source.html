136 !> Enables OBC module and reads configuration parameters
137 subroutine open_boundary_config(G, param_file, OBC)
138  type(dyn_horgrid_type), intent(in) :: g !< Ocean grid structure
139  type(param_file_type), intent(in) :: param_file !< Parameter file handle
140  type(ocean_obc_type), pointer :: obc !< Open boundary control structure
141  ! Local variables
142  integer :: l ! For looping over segments
143  character(len=15) :: segment_param_str ! The run-time parameter name for each segment
144  character(len=100) :: segment_str ! The contents (rhs) for parameter "segment_param_str"
145 
146  allocate(obc)
147 
148  call log_version(param_file, mod, version, "Controls where open boundaries are located, what "//&
149  "kind of boundary condition to impose, and what data to apply, if any.")
150  call get_param(param_file, mod, "OBC_NUMBER_OF_SEGMENTS", obc%number_of_segments, &
151  "The number of open boundary segments.", &
152  default=0)
153  call get_param(param_file, mod, "G_EARTH", obc%g_Earth, &
154  "The gravitational acceleration of the Earth.", &
155  units="m s-2", default = 9.80)
156  if (obc%number_of_segments > 0) then
157  ! Allocate everything
158  allocate(obc%OBC_segment_number(0:obc%number_of_segments))
159  do l=0,obc%number_of_segments
160  obc%OBC_segment_number(l)%Flather = .false.
161  obc%OBC_segment_number(l)%radiation = .false.
162  obc%OBC_segment_number(l)%oblique = .false.
163  obc%OBC_segment_number(l)%nudged = .false.
164  obc%OBC_segment_number(l)%specified = .false.
165  obc%OBC_segment_number(l)%gradient = .false.
166  obc%OBC_segment_number(l)%values_needed = .false.
167  obc%OBC_segment_number(l)%legacy = .false.
168  obc%OBC_segment_number(l)%direction = obc_none
169  obc%OBC_segment_number(l)%Tnudge_in = 0.0
170  obc%OBC_segment_number(l)%Tnudge_out = 0.0
171  enddo
172  allocate(obc%OBC_segment_u(g%IsdB:g%IedB,g%jsd:g%jed)) ; obc%OBC_segment_u(:,:) = obc_none
173  allocate(obc%OBC_segment_v(g%isd:g%ied,g%JsdB:g%JedB)) ; obc%OBC_segment_v(:,:) = obc_none
174 
175  do l = 1, obc%number_of_segments
176  write(segment_param_str(1:15),"('OBC_SEGMENT_',i3.3)") l
177  call get_param(param_file, mod, segment_param_str, segment_str, &
178  "Documentation needs to be dynamic?????", &
179  fail_if_missing=.true.)
180  segment_str = remove_spaces(segment_str)
181  if (segment_str(1:2) == 'I=') then
182  call setup_u_point_obc(obc, g, segment_str, l)
183  elseif (segment_str(1:2) == 'J=') then
184  call setup_v_point_obc(obc, g, segment_str, l)
185  else
186  call mom_error(fatal, "MOM_open_boundary.F90, open_boundary_config: "//&
187  "Unable to interpret "//segment_param_str//" = "//trim(segment_str))
188  endif
189  enddo
190  endif
191 
192  ! Safety check
193  if ((obc%Flather_u_BCs_exist_globally .or. obc%Flather_v_BCs_exist_globally) .and. &
194  .not.g%symmetric ) call mom_error(fatal, &
195  "MOM_open_boundary, open_boundary_config: "//&
196  "Symmetric memory must be used when using Flather OBCs.")
197 
198  if (.not.(obc%specified_u_BCs_exist_globally .or. obc%specified_v_BCs_exist_globally .or. &
199  obc%Flather_u_BCs_exist_globally .or. obc%Flather_v_BCs_exist_globally)) then
200  ! No open boundaries have been requested
201  call open_boundary_dealloc(obc)
202  endif
203 
204 end subroutine open_boundary_config

206 !> Parse an OBC_SEGMENT_%%% string starting with "I=" and configure placement and type of OBC accordingly
207 subroutine setup_u_point_obc(OBC, G, segment_str, l_seg)
208  type(ocean_obc_type), pointer :: OBC !< Open boundary control structure
209  type(dyn_horgrid_type), intent(in) :: G !< Ocean grid structure
210  character(len=*), intent(in) :: segment_str !< A string in form of "I=%,J=%:%,string"
211  integer, intent(in) :: l_seg !< which segment is this?
212  ! Local variables
213  integer :: I_obc, Js_obc, Je_obc ! Position of segment in global index space
214  integer :: j, this_kind, a_loop
215  character(len=32) :: action_str(5)
216 
217  ! This returns the global indices for the segment
218  call parse_segment_str(g%ieg, g%jeg, segment_str, i_obc, js_obc, je_obc, action_str )
219  i_obc = i_obc - g%idg_offset ! Convert to local tile indices on this tile
220  js_obc = js_obc - g%jdg_offset ! Convert to local tile indices on this tile
221  je_obc = je_obc - g%jdg_offset ! Convert to local tile indices on this tile
222  this_kind = obc_none
223 
224  ! Hack to extend segment by one point
225  if (js_obc<je_obc) then
226  js_obc = js_obc - 1 ; je_obc = je_obc + 1
227  else
228  js_obc = js_obc + 1 ; je_obc = je_obc - 1
229  endif
230 
231  if (je_obc>js_obc) obc%OBC_segment_number(l_seg)%direction = obc_direction_e
232  if (je_obc<js_obc) obc%OBC_segment_number(l_seg)%direction = obc_direction_w
233  do a_loop = 1,5
234  if (len_trim(action_str(a_loop)) == 0) then
235  cycle
236  elseif (trim(action_str(a_loop)) == 'FLATHER') then
237  this_kind = obc_flather
238  obc%OBC_segment_number(l_seg)%Flather = .true.
239  obc%Flather_u_BCs_exist_globally = .true.
240  elseif (trim(action_str(a_loop)) == 'ORLANSKI') then
241  obc%OBC_segment_number(l_seg)%radiation = .true.
242  obc%Flather_u_BCs_exist_globally = .true.
243  elseif (trim(action_str(a_loop)) == 'OBLIQUE') then
244  obc%OBC_segment_number(l_seg)%radiation = .true.
245  obc%OBC_segment_number(l_seg)%oblique = .true.
246  obc%Flather_u_BCs_exist_globally = .true.
247  elseif (trim(action_str(a_loop)) == 'NUDGED') then
248  obc%OBC_segment_number(l_seg)%nudged = .true.
249  obc%OBC_segment_number(l_seg)%values_needed = .true.
250  obc%OBC_segment_number(l_seg)%Tnudge_in = 1.0/(3*86400)
251  obc%OBC_segment_number(l_seg)%Tnudge_out = 1.0/(360*86400)
252  obc%nudged_u_BCs_exist_globally = .true.
253  elseif (trim(action_str(a_loop)) == 'GRADIENT') then
254  obc%OBC_segment_number(l_seg)%gradient = .true.
255  elseif (trim(action_str(a_loop)) == 'LEGACY') then
256  this_kind = obc_flather
257  obc%OBC_segment_number(l_seg)%legacy = .true.
258  obc%OBC_segment_number(l_seg)%Flather = .true.
259  obc%OBC_segment_number(l_seg)%radiation = .true.
260  obc%OBC_segment_number(l_seg)%gradient = .true.
261  obc%Flather_u_BCs_exist_globally = .true.
262  elseif (trim(action_str(a_loop)) == 'SIMPLE') then
263  obc%OBC_segment_number(l_seg)%specified = .true.
264  obc%OBC_segment_number(l_seg)%values_needed = .true.
265  obc%specified_u_BCs_exist_globally = .true. ! This avoids deallocation
266  ! Hack to undo the hack above for SIMPLE BCs
267  if (js_obc<je_obc) then
268  js_obc = js_obc + 1 ; je_obc = je_obc - 1
269  else
270  js_obc = js_obc - 1 ; je_obc = je_obc + 1
271  endif
272  else
273  call mom_error(fatal, "MOM_open_boundary.F90, setup_u_point_obc: "//&
274  "String '"//trim(action_str(a_loop))//"' not understood.")
275  endif
276 
277  if (i_obc<g%HI%IsdB .or. i_obc>g%HI%IedB) return ! Boundary is not on tile
278  if (max(js_obc,je_obc)<g%HI%JsdB .or. min(js_obc,je_obc)>g%HI%JedB) return ! Segment is not on tile
279 
280  do j=g%HI%jsd, g%HI%jed
281  if (j>min(js_obc,je_obc) .and. j<=max(js_obc,je_obc)) then
282  obc%OBC_segment_u(i_obc,j) = l_seg
283  if (je_obc>js_obc) then ! East is outward
284  if (this_kind == obc_flather) then
285  ! Set v points outside segment
286  if (obc%OBC_segment_v(i_obc+1,j) == obc_none) then
287  obc%OBC_segment_v(i_obc+1,j) = l_seg
288  endif
289  if (obc%OBC_segment_v(i_obc+1,j-1) == obc_none) then
290  obc%OBC_segment_v(i_obc+1,j-1) = l_seg
291  endif
292  endif
293  else ! West is outward
294  if (this_kind == obc_flather) then
295  ! Set v points outside segment
296  if (obc%OBC_segment_v(i_obc,j) == obc_none) then
297  obc%OBC_segment_v(i_obc,j) = l_seg
298  endif
299  if (obc%OBC_segment_v(i_obc,j-1) == obc_none) then
300  obc%OBC_segment_v(i_obc,j-1) = l_seg
301  endif
302  endif
303  endif
304  endif
305  enddo
306  enddo ! a_loop
307  obc%OBC_segment_number(l_seg)%Is_obc = i_obc
308  obc%OBC_segment_number(l_seg)%Ie_obc = i_obc
309  obc%OBC_segment_number(l_seg)%Js_obc = js_obc
310  obc%OBC_segment_number(l_seg)%Je_obc = je_obc
311 
312 end subroutine setup_u_point_obc

726  type(ocean_grid_type), intent(inout) :: g !< Ocean grid structure

755  dhdt = u_old(i-1,j,k)-u_new(i-1,j,k) !old-new
756  dhdx = u_new(i-1,j,k)-u_new(i-2,j,k) !in new time backward sasha for I-1
757  rx_new = 0.0
758  if (dhdt*dhdx > 0.0) rx_new = min( (dhdt/dhdx), rx_max)
759  rx_avg = (1.0-gamma_u)*obc%rx_old_u(i,j,k) + gamma_u*rx_new
760  obc%rx_old_u(i,j,k) = rx_avg
761  u_new(i,j,k) = (u_old(i,j,k) + rx_avg*u_new(i-1,j,k)) / (1.0+rx_avg)

764  ! dhdx = h_new(I,j,k)-h_new(I-1,j,k) !in new time
765  ! rx_new = 0.0

769  ! h_new(I+1,j,k) = (h_old(I+1,j,k) + rx_avg*h_new(I,j,k)) / (1.0+rx_avg) !original
770  elseif (obc%OBC_segment_number(obc%OBC_segment_u(i,j))%radiation) then
771  grad(i,j) = u_old(i,j+1,k) - u_old(i,j,k)
772  grad(i,j-1) = u_old(i,j,k) - u_old(i,j-1,k)
773  grad(i-1,j) = u_old(i-1,j+1,k) - u_old(i-1,j,k)

776  dhdx = u_new(i-1,j,k)-u_new(i-2,j,k) !in new time backward sasha for I-1
