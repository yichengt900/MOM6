24 !* *
25 !* By Alistair Adcroft and Robert Hallberg, 2010-2012 *
26 !* *
27 !* This file contains code that does the time-stepping of the *
28 !* adiabatic dynamic core, in this case with a pseudo-second order *
29 !* Runge-Kutta time stepping scheme for the momentum and a forward- *
30 !* backward coupling between the momentum and continuity equations, *
31 !* but without any splitting between the baroclinic and barotropic *
32 !* modes. Apart from the lack of splitting, this is closely analogous *
33 !* to the split time stepping scheme, and efforts have been taken to *
34 !* ensure that for certain configurations (e.g., very short *
35 !* baroclinic time steps, a single barotropic step per baroclinic *
36 !* step, and particular choices about how to coupled the baroclinic *
37 !* and barotropic solves, the two solutions reproduce each other. *
38 !* Although this time stepping scheme is not very efficient with a *
39 !* large number of layers, it is valuable for verifying the proper *
40 !* behavior of the more complicated split time stepping scheme, and *
41 !* is not too inefficient for use with only a few layers. *
42 !* *
43 !* The subroutine step_MOM_dyn_unsplit_RK2 actually does the time *
44 !* stepping, while register_restarts_dyn_unsplit_RK2 sets the fields *
45 !* that are found in a full restart file with this scheme, and *
46 !* initialize_dyn_unsplit_RK2 initializes the cpu clocks that are * *
47 !* used in this module. For largely historical reasons, this module *
48 !* does not have its own control structure, but shares the same *
49 !* control structure with MOM.F90 and the other MOM_dynamics_... *
50 !* modules. *
51 !* *
52 !* Macros written all in capital letters are defined in MOM_memory.h. *
53 !* *
54 !* A small fragment of the grid is shown below: *
55 !* *
56 !* j+1 x ^ x ^ x At x: q, CoriolisBu *
57 !* j+1 > o > o > At ^: v, PFv, CAv, vh, diffv, tauy, vbt, vhtr *
58 !* j x ^ x ^ x At >: u, PFu, CAu, uh, diffu, taux, ubt, uhtr *
59 !* j > o > o > At o: h, bathyT, eta, T, S, tr *
60 !* j-1 x ^ x ^ x *
61 !* i-1 i i+1 *
62 !* i i+1 *
63 !* *
64 !* The boundaries always run through q grid points (x). *
65 !* *

188 subroutine step_mom_dyn_unsplit_rk2(u_in, v_in, h_in, tv, visc, Time_local, dt, fluxes, &
189  p_surf_begin, p_surf_end, uh, vh, uhtr, vhtr, eta_av, G, GV, CS, &
190  VarMix, MEKE)
191  type(ocean_grid_type), intent(inout) :: g
192  type(verticalgrid_type), intent(in) :: gv
193  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(inout) :: u_in
194  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(inout) :: v_in
195  real, dimension(SZI_(G),SZJ_(G),SZK_(G)), intent(inout) :: h_in
196  type(thermo_var_ptrs), intent(in) :: tv
197  type(vertvisc_type), intent(inout) :: visc
198  type(time_type), intent(in) :: time_local
199  real, intent(in) :: dt
200  type(forcing), intent(in) :: fluxes
201  real, dimension(:,:), pointer :: p_surf_begin, p_surf_end
202  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(inout) :: uh
203  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(inout) :: vh
204  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)), intent(inout) :: uhtr
205  real, dimension(SZI_(G),SZJB_(G),SZK_(G)), intent(inout) :: vhtr
206  real, dimension(SZI_(G),SZJ_(G)), intent(out) :: eta_av
207  type(mom_dyn_unsplit_rk2_cs), pointer :: cs
208  type(varmix_cs), pointer :: varmix
209  type(meke_type), pointer :: meke
210 ! Arguments: u_in - The input and output zonal velocity, in m s-1.
211 ! (inout) v_in - The input and output meridional velocity, in m s-1.
212 ! (inout) h_in - The input and output layer thicknesses, in m or kg m-2,
213 ! depending on whether the Boussinesq approximation is made.
214 ! (in) tv - a structure pointing to various thermodynamic variables.
215 ! (inout) visc - A structure containing vertical viscosities, bottom drag
216 ! viscosities, and related fields.
217 ! (in) Time_local - The model time at the end of the time step.
218 ! (in) dt - The time step in s.
219 ! (in) fluxes - A structure containing pointers to any possible
220 ! forcing fields. Unused fields have NULL ptrs.
221 ! (in) p_surf_begin - A pointer (perhaps NULL) to the surface pressure
222 ! at the beginning of this dynamic step, in Pa.
223 ! (in) p_surf_end - A pointer (perhaps NULL) to the surface pressure
224 ! at the end of this dynamic step, in Pa.
225 ! (inout) uh - The zonal volume or mass transport, in m3 s-1 or kg s-1.
226 ! (inout) vh - The meridional volume or mass transport, in m3 s-1 or kg s-1.
227 ! (inout) uhtr - The accumulated zonal volume or mass transport since the last
228 ! tracer advection, in m3 or kg.
229 ! (inout) vhtr - The accumulated meridional volume or mass transport since the last
230 ! tracer advection, in m3 or kg.
231 ! (out) eta_av - The time-mean free surface height or column mass, in m or
232 ! kg m-2.
233 ! (in) G - The ocean's grid structure.
234 ! (in) GV - The ocean's vertical grid structure.
235 ! (in) CS - The control structure set up by initialize_dyn_unsplit_RK2.
236 ! (in) VarMix - A pointer to a structure with fields that specify the
237 ! spatially variable viscosities.
238 ! (inout) MEKE - A pointer to a structure containing fields related to
239 ! the Mesoscale Eddy Kinetic Energy.
240 
241  real, dimension(SZI_(G),SZJ_(G),SZK_(G)) :: h_av, hp
242  real, dimension(SZIB_(G),SZJ_(G),SZK_(G)) :: up
243  real, dimension(SZI_(G),SZJB_(G),SZK_(G)) :: vp
244  real, dimension(:,:), pointer :: p_surf
245  real :: dt_pred ! The time step for the predictor part of the baroclinic
246  ! time stepping.
247  logical :: dyn_p_surf
248  integer :: i, j, k, is, ie, js, je, isq, ieq, jsq, jeq, nz
249  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
250  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
251  dt_pred = dt * cs%BE
252 
253  h_av(:,:,:) = 0; hp(:,:,:) = 0
254  up(:,:,:) = 0
255  vp(:,:,:) = 0
256 
257  dyn_p_surf = associated(p_surf_begin) .and. associated(p_surf_end)
258  if (dyn_p_surf) then
259  call safe_alloc_ptr(p_surf,g%isd,g%ied,g%jsd,g%jed) ; p_surf(:,:) = 0.0
260  else
261  p_surf => fluxes%p_surf
262  endif
263 
264 ! Runge-Kutta second order accurate two step scheme is used to step
265 ! all of the fields except h. h is stepped separately.
266 
267  if (cs%debug) then
268  call mom_state_chksum("Start Predictor ", u_in, v_in, h_in, uh, vh, g, gv)
269  endif
270 
271 ! diffu = horizontal viscosity terms (u,h)
272  call enable_averaging(dt,time_local, cs%diag)
273  call cpu_clock_begin(id_clock_horvisc)
274  call horizontal_viscosity(u_in, v_in, h_in, cs%diffu, cs%diffv, meke, varmix, &
275  g, gv, cs%hor_visc_CSp)
276  call cpu_clock_end(id_clock_horvisc)
277  call disable_averaging(cs%diag)
278  call cpu_clock_begin(id_clock_pass)
279  call pass_vector(cs%diffu, cs%diffv, g%Domain)
280  call cpu_clock_end(id_clock_pass)
281 
282 ! This continuity step is solely for the Coroilis terms, specifically in the
283 ! denominator of PV and in the mass transport or PV.
284 ! uh = u[n-1]*h[n-1/2]
285 ! hp = h[n-1/2] + dt/2 div . uh
286  call cpu_clock_begin(id_clock_continuity)
287  ! This is a duplicate caclulation of the last continuity from the previous step
288  ! and could/should be optimized out. -AJA
289  call continuity(u_in, v_in, h_in, hp, uh, vh, dt_pred, g, gv, cs%continuity_CSp, &
290  obc=cs%OBC)
291  call cpu_clock_end(id_clock_continuity)
292  call cpu_clock_begin(id_clock_pass)
293  call pass_var(hp, g%Domain)
294  call pass_vector(uh, vh, g%Domain)
295  call cpu_clock_end(id_clock_pass)
296 
297 ! h_av = (h + hp)/2 (used in PV denominator)
298  call cpu_clock_begin(id_clock_mom_update)
299  do k=1,nz
300  do j=js-2,je+2 ; do i=is-2,ie+2
301  h_av(i,j,k) = (h_in(i,j,k) + hp(i,j,k)) * 0.5
302  enddo ; enddo ; enddo
303  call cpu_clock_end(id_clock_mom_update)
304 
305 ! CAu = -(f+zeta)/h_av vh + d/dx KE (function of u[n-1] and uh[n-1])
306  call cpu_clock_begin(id_clock_cor)
307  call coradcalc(u_in, v_in, h_av, uh, vh, cs%CAu, cs%CAv, cs%OBC, cs%ADp, &
308  g, gv, cs%CoriolisAdv_CSp)
309  call cpu_clock_end(id_clock_cor)
310 
311 ! PFu = d/dx M(h_av,T,S) (function of h[n-1/2])
312  call cpu_clock_begin(id_clock_pres)
313  if (dyn_p_surf) then ; do j=js-2,je+2 ; do i=is-2,ie+2
314  p_surf(i,j) = 0.5*p_surf_begin(i,j) + 0.5*p_surf_end(i,j)
315  enddo ; enddo ; endif
316  call pressureforce(h_in, tv, cs%PFu, cs%PFv, g, gv, &
317  cs%PressureForce_CSp, cs%ALE_CSp, p_surf)
318  call cpu_clock_end(id_clock_pres)
319  call cpu_clock_begin(id_clock_pass)
320  call pass_vector(cs%PFu, cs%PFv, g%Domain)
321  call pass_vector(cs%CAu, cs%CAv, g%Domain)
322  call cpu_clock_end(id_clock_pass)
323 
324  if (associated(cs%OBC)) then; if (cs%OBC%update_OBC) then
325  call update_obc_data(cs%OBC, g, h_in, time_local)
326  endif; endif
327 
328 ! up+[n-1/2] = u[n-1] + dt_pred * (PFu + CAu)
329  call cpu_clock_begin(id_clock_mom_update)
330  do k=1,nz ; do j=js,je ; do i=isq,ieq
331  up(i,j,k) = g%mask2dCu(i,j) * (u_in(i,j,k) + dt_pred * &
332  ((cs%PFu(i,j,k) + cs%CAu(i,j,k)) + cs%diffu(i,j,k)))
333  enddo ; enddo ; enddo
334  do k=1,nz ; do j=jsq,jeq ; do i=is,ie
335  vp(i,j,k) = g%mask2dCv(i,j) * (v_in(i,j,k) + dt_pred * &
336  ((cs%PFv(i,j,k) + cs%CAv(i,j,k)) + cs%diffv(i,j,k)))
337  enddo ; enddo ; enddo
338  call cpu_clock_end(id_clock_mom_update)
339 
340  if (cs%debug) &
341  call mom_accel_chksum("Predictor 1 accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv,&
342  cs%diffu, cs%diffv, g, gv)
343 
344 ! visc contains viscosity and BBL thickness (u_in,h_in)
345  if (visc%calc_bbl) then
346  call enable_averaging(visc%bbl_calc_time_interval, &
347  time_local+set_time(int(visc%bbl_calc_time_interval-dt)), cs%diag)
348  call set_viscous_bbl(u_in, v_in, h_av, tv, visc, g, gv, cs%set_visc_CSp)
349  call disable_averaging(cs%diag)
350  call cpu_clock_begin(id_clock_pass)
351  if (associated(visc%Ray_u) .and. associated(visc%Ray_v)) &
352  call pass_vector(visc%Ray_u, visc%Ray_v, g%Domain, &
353  to_all+scalar_pair, cgrid_ne)
354  if (associated(visc%kv_bbl_u) .and. associated(visc%kv_bbl_v)) then
355  call pass_vector(visc%bbl_thick_u, visc%bbl_thick_v, g%Domain, &
356  to_all+scalar_pair, cgrid_ne, complete=.false.)
357  call pass_vector(visc%kv_bbl_u, visc%kv_bbl_v, g%Domain, &
358  to_all+scalar_pair, cgrid_ne)
359  endif
360  call cpu_clock_end(id_clock_pass)
361  visc%calc_bbl = .false.
362  endif
363 
364  ! up[n-1/2] <- up*[n-1/2] + dt/2 d/dz visc d/dz up[n-1/2]
365  call cpu_clock_begin(id_clock_vertvisc)
366  call enable_averaging(dt, time_local, cs%diag)
367  call set_viscous_ml(up, vp, h_av, tv, fluxes, visc, dt_pred, g, gv, &
368  cs%set_visc_CSp)
369  call disable_averaging(cs%diag)
370  call vertvisc_coef(up, vp, h_av, fluxes, visc, dt_pred, g, gv, cs%vertvisc_CSp)
371  call vertvisc(up, vp, h_av, fluxes, visc, dt_pred, cs%OBC, cs%ADp, cs%CDp, &
372  g, gv, cs%vertvisc_CSp)
373  call cpu_clock_end(id_clock_vertvisc)
374  call cpu_clock_begin(id_clock_pass)
375  call pass_vector(up, vp, g%Domain)
376  call cpu_clock_end(id_clock_pass)
377 
378 ! uh = up[n-1/2] * h[n-1/2]
379 ! h_av = h + dt div . uh
380  call cpu_clock_begin(id_clock_continuity)
381  call continuity(up, vp, h_in, hp, uh, vh, &
382  dt, g, gv, cs%continuity_CSp, obc=cs%OBC)
383  call cpu_clock_end(id_clock_continuity)
384  call cpu_clock_begin(id_clock_pass)
385  call pass_var(hp, g%Domain)
386  call pass_vector(uh, vh, g%Domain)
387  call cpu_clock_end(id_clock_pass)
388 
389 ! h_av <- (h + hp)/2 (centered at n-1/2)
390  do k=1,nz ; do j=js-2,je+2 ; do i=is-2,ie+2
391  h_av(i,j,k) = (h_in(i,j,k) + hp(i,j,k)) * 0.5
392  enddo ; enddo ; enddo
393 
394  if (cs%debug) &
395  call mom_state_chksum("Predictor 1", up, vp, h_av, uh, vh, g, gv)
396 
397 ! CAu = -(f+zeta(up))/h_av vh + d/dx KE(up) (function of up[n-1/2], h[n-1/2])
398  call cpu_clock_begin(id_clock_cor)
399  call coradcalc(up, vp, h_av, uh, vh, cs%CAu, cs%CAv, cs%OBC, cs%ADp, &
400  g, gv, cs%CoriolisAdv_CSp)
401  call cpu_clock_end(id_clock_cor)
402 
403 ! call enable_averaging(dt,Time_local, CS%diag) ?????????????????????/
404 
405 ! up* = u[n] + (1+gamma) * dt * ( PFu + CAu ) Extrapolated for damping
406 ! u*[n+1] = u[n] + dt * ( PFu + CAu )
407  do k=1,nz ; do j=js,je ; do i=isq,ieq
408  up(i,j,k) = g%mask2dCu(i,j) * (u_in(i,j,k) + dt * (1.+cs%begw) * &
409  ((cs%PFu(i,j,k) + cs%CAu(i,j,k)) + cs%diffu(i,j,k)))
410  u_in(i,j,k) = g%mask2dCu(i,j) * (u_in(i,j,k) + dt * &
411  ((cs%PFu(i,j,k) + cs%CAu(i,j,k)) + cs%diffu(i,j,k)))
412  enddo ; enddo ; enddo

415  ((cs%PFv(i,j,k) + cs%CAv(i,j,k)) + cs%diffv(i,j,k)))

417  ((cs%PFv(i,j,k) + cs%CAv(i,j,k)) + cs%diffv(i,j,k)))
418  enddo ; enddo ; enddo

421 ! u[n] <- u*[n] + dt d/dz visc d/dz u[n]
422  call cpu_clock_begin(id_clock_vertvisc)
423  call vertvisc_coef(up, vp, h_av, fluxes, visc, dt, g, gv, cs%vertvisc_CSp)

431  call pass_vector(up, vp, g%Domain)
432  call pass_vector(u_in, v_in, g%Domain)
433  call cpu_clock_end(id_clock_pass)
434 
435 ! uh = up[n] * h[n] (up[n] might be extrapolated to damp GWs)
436 ! h[n+1] = h[n] + dt div . uh
437  call cpu_clock_begin(id_clock_continuity)
438  call continuity(up, vp, h_in, h_in, uh, vh, &
439  dt, g, gv, cs%continuity_CSp, obc=cs%OBC)
440  call cpu_clock_end(id_clock_continuity)
441  call cpu_clock_begin(id_clock_pass)
442  call pass_var(h_in, g%Domain)
443  call pass_vector(uh, vh, g%Domain)
444  call cpu_clock_end(id_clock_pass)
445 
446 ! Accumulate mass flux for tracer transport
447  do k=1,nz
448  do j=js-2,je+2 ; do i=isq-2,ieq+2
449  uhtr(i,j,k) = uhtr(i,j,k) + dt*uh(i,j,k)
450  enddo ; enddo
451  do j=jsq-2,jeq+2 ; do i=is-2,ie+2
452  vhtr(i,j,k) = vhtr(i,j,k) + dt*vh(i,j,k)
453  enddo ; enddo
454  enddo
455 
456  if (cs%debug) then
457  call mom_state_chksum("Corrector", u_in, v_in, h_in, uh, vh, g, gv)
458  call mom_accel_chksum("Corrector accel", cs%CAu, cs%CAv, cs%PFu, cs%PFv, &
459  cs%diffu, cs%diffv, g, gv)
460  endif
461 
462  if (gv%Boussinesq) then
463  do j=js,je ; do i=is,ie ; eta_av(i,j) = -g%bathyT(i,j) ; enddo ; enddo
464  else
465  do j=js,je ; do i=is,ie ; eta_av(i,j) = 0.0 ; enddo ; enddo
466  endif
467  do k=1,nz ; do j=js,je ; do i=is,ie
468  eta_av(i,j) = eta_av(i,j) + h_av(i,j,k)
469  enddo ; enddo ; enddo
470 
471  if (dyn_p_surf) deallocate(p_surf)
472 
473 ! Here various terms used in to update the momentum equations are
474 ! offered for averaging.
475  if (cs%id_PFu > 0) call post_data(cs%id_PFu, cs%PFu, cs%diag)
476  if (cs%id_PFv > 0) call post_data(cs%id_PFv, cs%PFv, cs%diag)
477  if (cs%id_CAu > 0) call post_data(cs%id_CAu, cs%CAu, cs%diag)
478  if (cs%id_CAv > 0) call post_data(cs%id_CAv, cs%CAv, cs%diag)
479 
480 ! Here the thickness fluxes are offered for averaging.
481  if (cs%id_uh > 0) call post_data(cs%id_uh, uh, cs%diag)
482  if (cs%id_vh > 0) call post_data(cs%id_vh, vh, cs%diag)
