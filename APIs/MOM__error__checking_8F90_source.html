83 subroutine check_redundant_v2d(mesg, u_comp, v_comp, G, is, ie, js, je, &
84  direction, stagger)
85  type(ocean_grid_type), intent(inout) :: G
86  character(len=*), intent(in) :: mesg
87  real, dimension(SZIB_(G),SZJ_(G)), intent(in) :: u_comp
88  real, dimension(SZI_(G),SZJB_(G)), intent(in) :: v_comp
89  integer, optional, intent(in) :: is, ie, js, je
90  integer, optional, intent(in) :: direction
91  integer, optional, intent(in) :: stagger
92 ! Arguments: u_comp - The u-component of the vector being checked.
93 ! (in) v_comp - The v-component of the vector being checked.
94 ! (in) mesg - A message indicating what is being checked.
95 ! (in) G - The ocean's grid structure.
96 ! (in/opt) is, ie, js, je - the i- and j- range of indices to check.
97 ! (in/opt) direction - the direction flag to be passed to pass_vector.
98 ! (in/opt) stagger - the stagger flag to be passed to pass_vector.
99 
100  real :: u_nonsym(SZI_(G),SZJ_(G))
101  real :: v_nonsym(SZI_(G),SZJ_(G))
102  real :: u_resym(SZIB_(G),SZJ_(G))
103  real :: v_resym(SZI_(G),SZJB_(G))
104  character(len=128) :: mesg2
105  type(group_pass_type), save :: pass_nonsym_uv, pass_resym_uv !For group halo pass
106 
107  integer :: i, j, is_ch, ie_ch, js_ch, je_ch
108  integer :: Isq, Ieq, Jsq, Jeq, isd, ied, jsd, jed, IsdB, IedB, JsdB, JedB
109  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
110  isd = g%isd ; ied = g%ied ; jsd = g%jsd ; jed = g%jed
111  isdb = g%IsdB ; iedb = g%IedB ; jsdb = g%JsdB ; jedb = g%JedB
112 
113  if ((isd == isdb) .and. (jsd == jsdb)) return
114 
115  do i=isd,ied ; do j=jsd,jed
116  u_nonsym(i,j) = u_comp(i,j) ; v_nonsym(i,j) = v_comp(i,j)
117  enddo ; enddo
118 
119  if (.not.associated(g%Domain_aux)) call mom_error(fatal," check_redundant"//&
120  " called with a non-associated auxiliary domain the grid type.")
121 
122  call create_group_pass(pass_nonsym_uv, u_nonsym, v_nonsym, g%Domain_aux, &
123  direction, stagger)
124  call create_group_pass(pass_resym_uv, u_resym, v_resym, g%Domain, &
125  direction, stagger)
126  call do_group_pass(pass_nonsym_uv, g%Domain_aux)
127 
128  do i=isdb,iedb ; do j=jsd,jed ; u_resym(i,j) = u_comp(i,j) ; enddo ; enddo
129  do i=isd,ied ; do j=jsdb,jedb ; v_resym(i,j) = v_comp(i,j) ; enddo ; enddo
130  do i=isd,ied ; do j=jsd,jed
131  u_resym(i,j) = u_nonsym(i,j) ; v_resym(i,j) = v_nonsym(i,j)
132  enddo ; enddo
133  call do_group_pass(pass_resym_uv, g%Domain)
134 
135  is_ch = isq ; ie_ch = ieq ; js_ch = jsq ; je_ch = jeq
136  if (present(is)) is_ch = is ; if (present(ie)) ie_ch = ie
137  if (present(js)) js_ch = js ; if (present(js)) je_ch = je
138 
139  do i=is_ch,ie_ch ; do j=js_ch+1,je_ch
140  if (u_resym(i,j) /= u_comp(i,j)) then
141  write(mesg2,'(" redundant u-components",2(1pe12.4)," differ by ", &
142  & 1pe12.4," at i,j = ",2i4," on pe ",i4)') &
143  u_comp(i,j), u_resym(i,j),u_comp(i,j)-u_resym(i,j),i,j,pe_here()
144  write(*,'(A130)') trim(mesg)//trim(mesg2)
145  endif
146  enddo ; enddo
147  do i=is_ch+1,ie_ch ; do j=js_ch,je_ch
148  if (v_resym(i,j) /= v_comp(i,j)) then
149  write(mesg2,'(" redundant v-comps",2(1pe12.4)," differ by ", &
150  & 1pe12.4," at i,j = ",2i4," x,y = ",2(1pe12.4)" on pe ",i4)') &
151  v_comp(i,j), v_resym(i,j),v_comp(i,j)-v_resym(i,j),i,j, &
152  g%geoLonBu(i,j), g%geoLatBu(i,j), pe_here()
153  write(*,'(A155)') trim(mesg)//trim(mesg2)
154  endif
155  enddo ; enddo
156 
157 end subroutine check_redundant_v2d

248 !> This function returns the sum over computational domain of all
249 !! processors of hThick*stuff, where stuff is a 3-d array at tracer points.
250 function totalstuff(HI, hThick, areaT, stuff)
251  type(hor_index_type), intent(in) :: hi !< A horizontal index type
252  real, dimension(HI%isd:,HI%jsd:,:), intent(in) :: hthick !< The array of thicknesses to use as weights
253  real, dimension(HI%isd:,HI%jsd:), intent(in) :: areat !< The array of cell areas in m2
254  real, dimension(HI%isd:,HI%jsd:,:), intent(in) :: stuff !< The array of stuff to be summed
255  real :: totalstuff
256  integer :: i, j, k, nz
257 
258  nz = size(hthick,3)
259  totalstuff = 0.
260  do k = 1, nz ; do j = hi%jsc, hi%jec ; do i = hi%isc, hi%iec
261  totalstuff = totalstuff + hthick(i,j,k) * stuff(i,j,k) * areat(i,j)
262  enddo ; enddo ; enddo
263  call sum_across_pes(totalstuff)
264 
265 end function totalstuff

269 !> This subroutine display the total thickness, temperature and salinity
270 !! as well as the change since the last call.
271 !! NOTE: This subroutine uses "save" data which is not thread safe and is purely
272 !! for extreme debugging without a proper debugger.
273 subroutine totaltands(HI, hThick, areaT, temperature, salinity, mesg)
274  type(hor_index_type), intent(in) :: hi !< A horizontal index type
275  real, dimension(HI%isd:,HI%jsd:,:), intent(in) :: hthick !< The array of thicknesses to use as weights
276  real, dimension(HI%isd:,HI%jsd:), intent(in) :: areat !< The array of cell areas in m2
277  real, dimension(HI%isd:,HI%jsd:,:), intent(in) :: temperature !< The temperature field to sum
278  real, dimension(HI%isd:,HI%jsd:,:), intent(in) :: salinity !< The salinity field to sum
279  character(len=*), intent(in) :: mesg !< An identifying message
280 
281  ! NOTE: This subroutine uses "save" data which is not thread safe and is purely for
282  ! extreme debugging without a proper debugger.
283  real, save :: totalh = 0., totalt = 0., totals = 0.
284 
285  logical, save :: firstcall = .true.
286  real :: thish, thist, thiss, delh, delt, dels
287  integer :: i, j, k, nz
288 
289  nz = size(hthick,3)
290  thish = 0.
291  do k = 1, nz ; do j = hi%jsc, hi%jec ; do i = hi%isc, hi%iec
292  thish = thish + hthick(i,j,k) * areat(i,j)
293  enddo ; enddo ; enddo
294  call sum_across_pes(thish)
295  thist = totalstuff(hi, hthick, areat, temperature)
296  thiss = totalstuff(hi, hthick, areat, salinity)
297 
298  if (is_root_pe()) then
299  if (firstcall) then
300  totalh = thish ; totalt = thist ; totals = thiss
301  write(0,*) 'Totals H,T,S:',thish,thist,thiss,' ',mesg
302  firstcall = .false.
303  else
304  delh = thish - totalh
305  delt = thist - totalt
306  dels = thiss - totals
307  totalh = thish ; totalt = thist ; totals = thiss
308  write(0,*) 'Tot/del H,T,S:',thish,thist,thiss,delh,delt,dels,' ',mesg
309  endif
310  endif
311 
312 end subroutine totaltands
