781 !------------------------------------------------------------------------------
782 ! Dispatching regridding routine: regridding & remapping
783 !------------------------------------------------------------------------------
784 subroutine regridding_main( remapCS, CS, G, GV, h, tv, h_new, dzInterface, frac_shelf_h)
785 !------------------------------------------------------------------------------
786 ! This routine takes care of (1) building a new grid and (2) remapping between
787 ! the old grid and the new grid. The creation of the new grid can be based
788 ! on z coordinates, target interface densities, sigma coordinates or any
789 ! arbitrary coordinate system.
790 ! The MOM6 interface positions are always calculated from the bottom up by
791 ! accumulating the layer thicknesses starting at z=-G%bathyT. z increases
792 ! upwards (decreasing k-index).
793 ! The new grid is defined by the change in position of those interfaces in z
794 ! dzInterface = zNew - zOld.
795 ! Thus, if the regridding inflates the top layer, hNew(1) > hOld(1), then the
796 ! second interface moves downward, zNew(2) < zOld(2), and dzInterface(2) < 0.
797 ! hNew(k) = hOld(k) - dzInterface(k+1) + dzInterface(k)
798 ! IMPORTANT NOTE:
799 ! This is the converse of the sign convention used in the remapping code!
800 !------------------------------------------------------------------------------
801 
802  ! Arguments
803  type(remapping_cs), intent(in) :: remapcs !< Remapping parameters and options
804  type(regridding_cs), intent(in) :: cs !< Regridding control structure
805  type(ocean_grid_type), intent(in) :: g !< Ocean grid structure
806  type(verticalgrid_type), intent(in) :: gv !< Ocean vertical grid structure
807  real, dimension(SZI_(G),SZJ_(G), SZK_(GV)), intent(inout) :: h !< Current 3D grid obtained after the last time step
808  type(thermo_var_ptrs), intent(inout) :: tv !< Thermodynamical variables (T, S, ...)
809  real, dimension(SZI_(G),SZJ_(G), SZK_(GV)), intent(inout) :: h_new !< New 3D grid consistent with target coordinate
810  real, dimension(SZI_(G),SZJ_(G), SZK_(GV)+1), intent(inout) :: dzinterface !< The change in position of each interface
811  real, dimension(:,:), optional, pointer :: frac_shelf_h !< Fractional ice shelf coverage
812  ! Local variables
813  real :: trickgnucompiler
814  logical :: use_ice_shelf
815 
816  use_ice_shelf = .false.
817  if (present(frac_shelf_h)) then
818  if (associated(frac_shelf_h)) use_ice_shelf = .true.
819  endif
820 
821  select case ( cs%regridding_scheme )
822 
823  case ( regridding_zstar )
824  if (use_ice_shelf) then
825  call build_zstar_grid( cs, g, gv, h, dzinterface, frac_shelf_h )
826  else
827  call build_zstar_grid( cs, g, gv, h, dzinterface )
828  endif
829  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
830 
831  case ( regridding_sigma_shelf_zstar)
832  call build_zstar_grid( cs, g, gv, h, dzinterface )
833  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
834 
835  case ( regridding_sigma )
836  call build_sigma_grid( cs, g, gv, h, dzinterface )
837  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
838 
839  case ( regridding_rho )
840  call convective_adjustment(g, gv, h, tv)
841  call build_rho_grid( g, gv, h, tv, dzinterface, remapcs, cs )
842  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
843 
844  case ( regridding_arbitrary )
845  call build_grid_arbitrary( g, gv, h, dzinterface, trickgnucompiler, cs )
846  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
847 
848  case ( regridding_hycom1 )
849  call build_grid_hycom1( g, gv, h, tv, dzinterface, remapcs, cs )
850  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
851 
852  case ( regridding_slight )
853  call build_grid_slight( g, gv, h, tv, dzinterface, remapcs, cs )
854  call calc_h_new_by_dz(g, gv, h, dzinterface, h_new)
855 
856  case default
857  call mom_error(fatal,'MOM_regridding, regridding_main: '//&
858  'Unknown regridding scheme selected!')
859 
860  end select ! type of grid
861 
862 #ifdef __DO_SAFETY_CHECKS__
863  call check_remapping_grid(g, gv, h, dzinterface,'in regridding_main')
864 #endif
865 
866 end subroutine regridding_main

868 !> Calculates h_new from h + delta_k dzInterface
869 subroutine calc_h_new_by_dz(G, GV, h, dzInterface, h_new)
870  type(ocean_grid_type), intent(in) :: G !< Grid structure
871  type(verticalgrid_type), intent(in) :: GV !< Ocean vertical grid structure
872  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)), intent(in) :: h !< Old layer thicknesses (m)
873  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)+1), intent(in) :: dzInterface !< Change in interface positions (m)
874  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)), intent(inout) :: h_new !< New layer thicknesses (m)
875  ! Local variables
876  integer :: i, j, k
877 
878 !$OMP parallel do default(none) shared(G,GV,h,dzInterface,h_new)
879  do j = g%jsc-1,g%jec+1
880  do i = g%isc-1,g%iec+1
881  if (g%mask2dT(i,j)>0.) then
882  do k=1,gv%ke
883  h_new(i,j,k) = max( 0., h(i,j,k) + ( dzinterface(i,j,k) - dzinterface(i,j,k+1) ) )
884  enddo
885  else
886  h_new(i,j,:) = h(i,j,:)
887  endif
888  enddo
889  enddo
890 
891 end subroutine calc_h_new_by_dz

897  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)), intent(in) :: h !< Layer thicknesses (m)
898  real, dimension(SZI_(G),SZJ_(G),SZK_(GV)+1), intent(in) :: dzinterface !< Change in interface positions (m)

969 !> Returns the change in interface position motion after filtering and
970 !! assuming the top and bottom interfaces do not move. The filtering is
971 !! a function of depth, and is applied as the integrated average filtering
972 !! over the trajectory of the interface. By design, this code can not give
973 !! tangled interfaces provided that z_old and z_new are not already tangled.
974 subroutine filtered_grid_motion( CS, nk, z_old, z_new, dz_g )
975  type(regridding_cs), intent(in) :: CS !< Regridding control structure
976  integer, intent(in) :: nk !< Number of cells
977  real, dimension(nk+1), intent(in) :: z_old !< Old grid position (m)
978  real, dimension(nk+1), intent(in) :: z_new !< New grid position (m)
979  real, dimension(nk+1), intent(inout) :: dz_g !< Change in interface positions (m)
980  ! Local variables
981  real :: sgn ! The sign convention for downward.
982  real :: dz_tgt, zr1
983  real :: Aq, Bq, dz0, z0, F0
984  real :: zs, zd, dzwt, Idzwt
985  real :: wtd, Iwtd
986  real :: Int_zs, Int_zd, dInt_zs_zd
987 ! For debugging:
988  real, dimension(nk+1) :: z_act
989 ! real, dimension(nk+1) :: ddz_g_s, ddz_g_d
990  logical :: debug = .false.
991  integer :: k
992 
993  if ((z_old(nk+1) - z_old(1)) * (z_new(nk+1) - z_new(1)) < 0.0) then
994  call mom_error(fatal, "filtered_grid_motion: z_old and z_new use different sign conventions.")
995  elseif ((z_old(nk+1) - z_old(1)) * (z_new(nk+1) - z_new(1)) == 0.0) then
996  ! This is a massless column, so do nothing and return.
997  do k=1,nk+1 ; dz_g(k) = 0.0 ; enddo ; return
998  elseif ((z_old(nk+1) - z_old(1)) + (z_new(nk+1) - z_new(1)) > 0.0) then
999  sgn = 1.0
1000  else
1001  sgn = -1.0
1002  endif
1003 
1004  if (debug) then
1005  do k=2,nk+1
1006  if (sgn*(z_new(k)-z_new(k-1)) < -5e-16*(abs(z_new(k))+abs(z_new(k-1))) ) &
1007  call mom_error(fatal, "filtered_grid_motion: z_new is tangled.")
1008  if (sgn*(z_old(k)-z_old(k-1)) < -5e-16*(abs(z_old(k))+abs(z_old(k-1))) ) &
1009  call mom_error(fatal, "filtered_grid_motion: z_old is tangled.")
1010  enddo
1011  ! ddz_g_s(:) = 0.0 ; ddz_g_d(:) = 0.0
1012  endif
1013 
1014  zs = cs%depth_of_time_filter_shallow
1015  zd = cs%depth_of_time_filter_deep
1016  wtd = 1.0 - cs%old_grid_weight
1017  iwtd = 1.0 / wtd
1018 
1019  dzwt = (zd - zs)
1020  idzwt = 0.0 ; if (abs(zd - zs) > 0.0) idzwt = 1.0 / (zd - zs)
1021  dint_zs_zd = 0.5*(1.0 + iwtd) * (zd - zs)
1022  aq = 0.5*(iwtd - 1.0)
1023 
1024  dz_g(1) = 0.0
1025  do k = 2,nk
1026  ! zr1 is positive and increases with depth, and dz_tgt is positive downward.
1027  dz_tgt = sgn*(z_new(k) - z_old(k))
1028  zr1 = sgn*(z_old(k) - z_old(1))
1029 
1030  ! First, handle the two simple and common cases that do not pass through
1031  ! the adjustment rate transition zone.
1032  if ((zr1 > zd) .and. (zr1 + wtd * dz_tgt > zd)) then
1033  dz_g(k) = sgn * wtd * dz_tgt
1034  elseif ((zr1 < zs) .and. (zr1 + dz_tgt < zs)) then
1035  dz_g(k) = sgn * dz_tgt
1036  else
1037  ! Find the new value by inverting the equation
1038  ! integral(0 to dz_new) Iwt(z) dz = dz_tgt
1039  ! This is trivial where Iwt is a constant, and agrees with the two limits above.
1040 
1041  ! Take test values at the transition points to figure out which segment
1042  ! the new value will be found in.
1043  if (zr1 >= zd) then
1044  int_zd = iwtd*(zd - zr1)
1045  int_zs = int_zd - dint_zs_zd
1046  elseif (zr1 <= zs) then
1047  int_zs = (zs - zr1)
1048  int_zd = dint_zs_zd + (zs - zr1)
1049  else
1050 ! Int_zd = (zd - zr1) * (Iwtd + 0.5*(1.0 - Iwtd) * (zd - zr1) / (zd - zs))
1051  int_zd = (zd - zr1) * (iwtd*(0.5*(zd+zr1) - zs) + 0.5*(zd - zr1)) * idzwt
1052  int_zs = (zs - zr1) * (0.5*iwtd * ((zr1 - zs)) + (zd - 0.5*(zr1+zs))) * idzwt
1053  ! It has been verified that Int_zs = Int_zd - dInt_zs_zd to within roundoff.
1054  endif
1055 
1056  if (dz_tgt >= int_zd) then ! The new location is in the deep, slow region.
1057  dz_g(k) = sgn * ((zd-zr1) + wtd*(dz_tgt - int_zd))
1058  elseif (dz_tgt <= int_zs) then ! The new location is in the shallow region.
1059  dz_g(k) = sgn * ((zs-zr1) + (dz_tgt - int_zs))
1060  else ! We need to solve a quadratic equation for z_new.
1061  ! For accuracy, do the integral from the starting depth or the nearest
1062  ! edge of the transition region. The results with each choice are
1063  ! mathematically equivalent, but differ in roundoff, and this choice
1064  ! should minimize the likelihood of inadvertently overlapping interfaces.
1065  if (zr1 <= zs) then ; dz0 = zs-zr1 ; z0 = zs ; f0 = dz_tgt - int_zs
1066  elseif (zr1 >= zd) then ; dz0 = zd-zr1 ; z0 = zd ; f0 = dz_tgt - int_zd
1067  else ; dz0 = 0.0 ; z0 = zr1 ; f0 = dz_tgt ; endif
1068 
1069  bq = (dzwt + 2.0*aq*(z0-zs))
1070  ! Solve the quadratic: Aq*(zn-z0)**2 + Bq*(zn-z0) - F0*dzwt = 0
1071  ! Note that b>=0, and the two terms in the standard form cancel for the right root.
1072  dz_g(k) = sgn * (dz0 + 2.0*f0*dzwt / (bq + sqrt(bq**2 + 4.0*aq*f0*dzwt) ))
1073 
1074 ! if (debug) then
1075 ! dz0 = zs-zr1 ; z0 = zs ; F0 = dz_tgt - Int_zs ; Bq = (dzwt + 2.0*Aq*(z0-zs))
1076 ! ddz_g_s(k) = sgn * (dz0 + 2.0*F0*dzwt / (Bq + sqrt(Bq**2 + 4.0*Aq*F0*dzwt) )) - dz_g(k)
1077 ! dz0 = zd-zr1 ; z0 = zd ; F0 = dz_tgt - Int_zd ; Bq = (dzwt + 2.0*Aq*(z0-zs))
1078 ! ddz_g_d(k) = sgn * (dz0 + 2.0*F0*dzwt / (Bq + sqrt(Bq**2 + 4.0*Aq*F0*dzwt) )) - dz_g(k)
1079 !
1080 ! if (abs(ddz_g_s(k)) > 1e-12*(abs(dz_g(k)) + abs(dz_g(k)+ddz_g_s(k)))) &
1081 ! call MOM_error(WARNING, "filtered_grid_motion: Expect z_output to be tangled (sc).")
1082 ! if (abs(ddz_g_d(k) - ddz_g_s(k)) > 1e-12*(abs(dz_g(k)+ddz_g_d(k)) + abs(dz_g(k)+ddz_g_s(k)))) &
1083 ! call MOM_error(WARNING, "filtered_grid_motion: Expect z_output to be tangled.")
1084 ! endif
1085  endif
1086 
1087  endif
1088  enddo
1089  dz_g(nk+1) = 0.0
1090 
1091  if (debug) then
1092  do k=1,nk+1 ; z_act(k) = z_old(k) + dz_g(k) ; enddo
1093  do k=2,nk+1
1094  if (sgn*((z_act(k))-z_act(k-1)) < -1e-15*(abs(z_act(k))+abs(z_act(k-1))) ) &
1095  call mom_error(fatal, "filtered_grid_motion: z_output is tangled.")
1096  enddo
1097  endif
1098 
1099 end subroutine filtered_grid_motion

1192 !> Builds a z* coordinate with a minimum thickness
1193 subroutine build_zstar_column(CS, nz, depth, total_thickness, zInterface, z_rigid_top, eta_orig)
1194  type(regridding_cs), intent(in) :: cs !< Regridding control structure
1195  integer, intent(in) :: nz !< Number of levels
1196  real, intent(in) :: depth !< Depth of ocean bottom (positive in m)
1197  real, intent(in) :: total_thickness !< Column thickness (positive in m)
1198  real, dimension(nz+1), intent(inout) :: zinterface !< Absolute positions of interfaces
1199  real, optional, intent(in) :: z_rigid_top !< The height of a rigid top (negative in m)
1200  real, optional, intent(in) :: eta_orig !< The actual original height of the top (m)
1201  ! Local variables
1202  real :: eta, stretching, dh, min_thickness, z0_top, z_star
1203  integer :: k
1204  logical :: new_zstar_def
1205 
1206  new_zstar_def = .false.
1207  min_thickness = min( cs%min_thickness, total_thickness/real(nz) )
1208  z0_top = 0.
1209  if (present(z_rigid_top)) then
1210  z0_top = z_rigid_top
1211  new_zstar_def = .true.
1212  endif
1213 
1214  ! Position of free-surface (or the rigid top, for which eta ~ z0_top)
1215  eta = total_thickness - depth
1216  if (present(eta_orig)) eta = eta_orig
1217 
1218  ! Conventional z* coordinate:
1219  ! z* = (z-eta) / stretching where stretching = (H+eta)/H
1220  ! z = eta + stretching * z*
1221  ! The above gives z*(z=eta) = 0, z*(z=-H) = -H.
1222  ! With a rigid top boundary at eta = z0_top then
1223  ! z* = z0 + (z-eta) / stretching where stretching = (H+eta)/(H+z0)
1224  ! z = eta + stretching * (z*-z0) * stretching
1225  stretching = total_thickness / ( depth + z0_top )
1226 
1227  if (new_zstar_def) then
1228  ! z_star is the notional z* coordinate in absence of upper/lower topography
1229  z_star = 0. ! z*=0 at the free-surface
1230  zinterface(1) = eta ! The actual position of the top of the column
1231  do k = 2,nz
1232  z_star = z_star - cs%coordinateResolution(k-1)
1233  ! This ensures that z is below a rigid upper surface (ice shelf bottom)
1234  zinterface(k) = min( eta + stretching * ( z_star - z0_top ), z0_top )
1235  ! This ensures that the layer in inflated
1236  zinterface(k) = min( zinterface(k), zinterface(k-1) - min_thickness )
1237  ! This ensures that z is above or at the topography
1238  zinterface(k) = max( zinterface(k), -depth + real(nz+1-k) * min_thickness )
1239  enddo
1240  zinterface(nz+1) = -depth
1241 
1242  else
1243  ! Integrate down from the top for a notional new grid, ignoring topography
1244  ! The starting position is offset by z0_top which, if z0_top<0, will place
1245  ! interfaces above the rigid boundary.
1246  zinterface(1) = eta
1247  do k = 1,nz
1248  dh = stretching * cs%coordinateResolution(k) ! Notional grid spacing
1249  zinterface(k+1) = zinterface(k) - dh
1250  enddo
1251 
1252  ! Integrating up from the bottom adjusting interface position to accommodate
1253  ! inflating layers without disturbing the interface above
1254  zinterface(nz+1) = -depth
1255  do k = nz,1,-1
1256  if ( zinterface(k) < (zinterface(k+1) + min_thickness) ) then
1257  zinterface(k) = zinterface(k+1) + min_thickness
1258  endif
1259  enddo
1260  endif
1261 
1262 end subroutine build_zstar_column

1608  h0(:) = h1(:)
1609

1621  zinterface(nz+1) = -depth
1622  do k = nz,1,-1
1623  zinterface(k) = zinterface(k+1) + h1(k)

2724 !------------------------------------------------------------------------------
2725 ! Given target value, find corresponding coordinate for given polynomial
2726 !------------------------------------------------------------------------------
2727 function get_polynomial_coordinate ( N, h, x_g, ppoly_E, ppoly_coefficients, &
2728  target_value, degree ) result ( x_tgt )
2729 ! ------------------------------------------------------------------------------
2730 ! Here, 'ppoly' is assumed to be a piecewise discontinuous polynomial of degree
2731 ! 'degree' throughout the domain defined by 'grid'. A target value is given
2732 ! and we need to determine the corresponding grid coordinate to define the
2733 ! new grid.
2734 !
2735 ! If the target value is out of range, the grid coordinate is simply set to
2736 ! be equal to one of the boundary coordinates, which results in vanished layers
2737 ! near the boundaries.
2738 !
2739 ! IT IS ASSUMED THAT THE PIECEWISE POLYNOMIAL IS MONOTONICALLY INCREASING.
2740 ! IF THIS IS NOT THE CASE, THE NEW GRID MAY BE ILL-DEFINED.
2741 !
2742 ! It is assumed that the number of cells defining 'grid' and 'ppoly' are the
2743 ! same.
2744 ! ------------------------------------------------------------------------------
2745 
2746  ! Arguments
2747  integer, intent(in) :: n ! The number of grid cells
2748  real, dimension(:), intent(in) :: h ! Grid cell thicknesses (size N)
2749  real, dimension(:), intent(in) :: x_g ! Grid interface locations (size N+1)
2750  real, dimension(:,:), intent(in) :: ppoly_e !Edge value of polynomial
2751  real, dimension(:,:), intent(in) :: ppoly_coefficients !Coefficients of polynomial
2752  real, intent(in) :: target_value
2753  integer, intent(in) :: degree ! The degree of the polynomials
2754 
2755  real :: x_tgt !< The position of x_g at which target_value is found.
2756 
2757  ! Local variables
2758  integer :: i, k ! loop indices
2759  integer :: k_found ! index of target cell
2760  integer :: iter
2761  real :: xi0 ! normalized target coordinate
2762  real, dimension(5) :: a ! polynomial coefficients
2763  real :: numerator
2764  real :: denominator
2765  real :: delta ! Newton-Raphson increment
2766  real :: x ! global target coordinate
2767  real :: eps ! offset used to get away from
2768  ! boundaries
2769  real :: grad ! gradient during N-R iterations
2770 
2771  eps = nr_offset
2772 
2773  k_found = -1
2774 
2775  ! If the target value is outside the range of all values, we
2776  ! force the target coordinate to be equal to the lowest or
2777  ! largest value, depending on which bound is overtaken
2778  if ( target_value <= ppoly_e(1,1) ) then
2779  x_tgt = x_g(1)
2780  return ! return because there is no need to look further
2781  end if
2782 
2783  ! Since discontinuous edge values are allowed, we check whether the target
2784  ! value lies between two discontinuous edge values at interior interfaces
2785  do k = 2,n
2786  if ( ( target_value >= ppoly_e(k-1,2) ) .AND. &
2787  ( target_value <= ppoly_e(k,1) ) ) then
2788  x_tgt = x_g(k)
2789  return ! return because there is no need to look further
2790  exit
2791  end if
2792  end do
2793 
2794  ! If the target value is outside the range of all values, we
2795  ! force the target coordinate to be equal to the lowest or
2796  ! largest value, depending on which bound is overtaken
2797  if ( target_value >= ppoly_e(n,2) ) then
2798  x_tgt = x_g(n+1)
2799  return ! return because there is no need to look further
2800  end if
2801 
2802  ! At this point, we know that the target value is bounded and does not
2803  ! lie between discontinuous, monotonic edge values. Therefore,
2804  ! there is a unique solution. We loop on all cells and find which one
2805  ! contains the target value. The variable k_found holds the index value
2806  ! of the cell where the taregt value lies.
2807  do k = 1,n
2808  if ( ( target_value > ppoly_e(k,1) ) .AND. &
2809  ( target_value < ppoly_e(k,2) ) ) then
2810  k_found = k
2811  exit
2812  end if
2813  end do
2814 
2815  ! At this point, 'k_found' should be strictly positive. If not, this is
2816  ! a major failure because it means we could not find any target cell
2817  ! despite the fact that the target value lies between the extremes. It
2818  ! means there is a major problem with the interpolant. This needs to be
2819  ! reported.
2820  if ( k_found == -1 ) then
2821  write(*,*) target_value, ppoly_e(1,1), ppoly_e(n,2)
2822  write(*,*) 'Could not find target coordinate in ' //&
2823  '"get_polynomial_coordinate". This is caused by an '//&
2824  'inconsistent interpolant (perhaps not monotonically '//&
2825  'increasing)'
2826  call mom_error( fatal, 'Aborting execution' )
2827  end if
2828 
2829  ! Reset all polynomial coefficients to 0 and copy those pertaining to
2830  ! the found cell
2831  a(:) = 0.0
2832  do i = 1,degree+1
2833  a(i) = ppoly_coefficients(k_found,i)
2834  end do
2835 
2836  ! Guess value to start Newton-Raphson iterations (middle of cell)
2837  xi0 = 0.5
2838  iter = 1
2839  delta = 1e10
2840 
2841  ! Newton-Raphson iterations
2842  do
2843  if ( ( iter > nr_iterations ) .OR. &
2844  ( abs(delta) < nr_tolerance ) ) then
2845  exit
2846  end if
2847 
2848  numerator = a(1) + a(2)*xi0 + a(3)*xi0*xi0 + a(4)*xi0*xi0*xi0 + &
2849  a(5)*xi0*xi0*xi0*xi0 - target_value
2850 
2851  denominator = a(2) + 2*a(3)*xi0 + 3*a(4)*xi0*xi0 + 4*a(5)*xi0*xi0*xi0
2852 
2853  delta = - ( numerator ) / &
2854  ( denominator )
2855 
2856  xi0 = xi0 + delta
2857 
2858  ! Check whether new estimate is out of bounds. If the new estimate is
2859  ! indeed out of bounds, we manually set it to be equal to the overtaken
2860  ! bound with a small offset towards the interior when the gradient of
2861  ! the function at the boundary is zero (in which case, the Newton-Raphson
2862  ! algorithm does not converge).
2863  if ( xi0 < 0.0 ) then
2864  xi0 = 0.0
2865  grad = a(2)
2866  if ( grad == 0.0 ) xi0 = xi0 + eps
2867  end if
2868 
2869  if ( xi0 > 1.0 ) then
2870  xi0 = 1.0
2871  grad = a(2) + 2*a(3) + 3*a(4) + 4*a(5)
2872  if ( grad == 0.0 ) xi0 = xi0 - eps
2873  end if
2874 
2875  iter = iter + 1
2876 
2877  end do ! end Newton-Raphson iterations
2878 
2879  x_tgt = x_g(k_found) + xi0 * h(k_found)
2880 
2881 end function get_polynomial_coordinate

3413 !!

3421 !! \f]
