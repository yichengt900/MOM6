566 !> Adjust interface heights to fit the bathymetry and diagnose layer thickness.
567 !! If the bottom most interface is below the topography then the bottom-most
568 !! layers are contracted to GV%Angstrom_z.
569 !! If the bottom most interface is above the topography then the entire column
570 !! is dilated (expanded) to fill the void.
571 !! @remark{There is a (hard-wired) "tolerance" parameter such that the
572 !! criteria for adjustment must equal or exceed 10cm.}
573 !! @param[in] G Grid type
574 !! @param[in,out] eta Interface heights
575 !! @param[out] h Layer thicknesses
576 subroutine adjustetatofitbathymetry(G, GV, eta, h)
577  type(ocean_grid_type), intent(in) :: G
578  type(verticalgrid_type), intent(in) :: GV
579  real, dimension(SZI_(G),SZJ_(G), SZK_(G)+1), intent(inout) :: eta
580  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(inout) :: h
581  ! Local variables
582  integer :: i, j, k, is, ie, js, je, nz, contractions, dilations
583  real, parameter :: hTolerance = 0.1 !< Tolerance to exceed adjustment criteria (m)
584  real :: hTmp, eTmp, dilate
585  character(len=100) :: mesg
586 
587  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
588 
589  contractions = 0
590  do j=js,je ; do i=is,ie
591  if (-eta(i,j,nz+1) > g%bathyT(i,j) + htolerance) then
592  eta(i,j,nz+1) = -g%bathyT(i,j)
593  contractions = contractions + 1
594  endif
595  enddo ; enddo
596  call sum_across_pes(contractions)
597  if ((contractions > 0) .and. (is_root_pe())) then
598  write(mesg,'("Thickness initial conditions were contracted ",'// &
599  '"to fit topography in ",I8," places.")') contractions
600  call mom_error(warning, 'adjustEtaToFitBathymetry: '//mesg)
601  endif
602 
603  do k=nz,1,-1 ; do j=js,je ; do i=is,ie
604  ! Collapse layers to thinnest possible if the thickness less than
605  ! the thinnest possible (or negative).
606  if (eta(i,j,k) < (eta(i,j,k+1) + gv%Angstrom_z)) then
607  eta(i,j,k) = eta(i,j,k+1) + gv%Angstrom_z
608  h(i,j,k) = gv%Angstrom_z
609  else
610  h(i,j,k) = eta(i,j,k) - eta(i,j,k+1)
611  endif
612  enddo ; enddo ; enddo
613 
614  dilations = 0
615  do j=js,je ; do i=is,ie
616  ! The whole column is dilated to accommodate deeper topography than
617  ! the bathymetry would indicate.
618  ! This should be... if ((G%mask2dt(i,j)*(eta(i,j,1)-eta(i,j,nz+1)) > 0.0) .and. &
619  if (-eta(i,j,nz+1) < g%bathyT(i,j) - htolerance) then
620  dilations = dilations + 1
621  if (eta(i,j,1) <= eta(i,j,nz+1)) then
622  do k=1,nz ; h(i,j,k) = (eta(i,j,1)+g%bathyT(i,j)) / real(nz) ; enddo
623  else
624  dilate = (eta(i,j,1)+g%bathyT(i,j)) / (eta(i,j,1)-eta(i,j,nz+1))
625  do k=1,nz ; h(i,j,k) = h(i,j,k) * dilate ; enddo
626  endif
627  do k=nz, 2, -1; eta(i,j,k) = eta(i,j,k+1) + h(i,j,k); enddo
628  endif
629  enddo ; enddo
630  call sum_across_pes(dilations)
631  if ((dilations > 0) .and. (is_root_pe())) then
632  write(mesg,'("Thickness initial conditions were dilated ",'// &
633  '"to fit topography in ",I8," places.")') dilations
634  call mom_error(warning, 'adjustEtaToFitBathymetry: '//mesg)
635  endif

637 end subroutine adjustetatofitbathymetry

701 subroutine convert_thickness(h, G, GV, param_file, tv)
702  type(ocean_grid_type), intent(in) :: G
703  type(verticalgrid_type), intent(in) :: GV
704  real, dimension(SZI_(G),SZJ_(G), SZK_(G)), intent(inout) :: h
705  type(param_file_type), intent(in) :: param_file
706  type(thermo_var_ptrs), intent(in) :: tv
707 ! Arguments: h - The thickness that is being initialized.
708 ! (in) G - The ocean's grid structure.
709 ! (in) GV - The ocean's vertical grid structure.
710 ! (in) param_file - A structure indicating the open file to parse for
711 ! model parameter values.
712  real, dimension(SZI_(G),SZJ_(G)) :: &
713  p_top, p_bot
714  real :: dz_geo(SZI_(G),SZJ_(G)) ! The change in geopotential height
715  ! across a layer, in m2 s-2.
716  real :: rho(SZI_(G))
717  real :: I_gEarth
718  logical :: Boussinesq
719  integer :: i, j, k, is, ie, js, je, Isq, Ieq, Jsq, Jeq, nz
720  integer :: itt, max_itt
721 
722  is = g%isc ; ie = g%iec ; js = g%jsc ; je = g%jec ; nz = g%ke
723  isq = g%IscB ; ieq = g%IecB ; jsq = g%JscB ; jeq = g%JecB
724  max_itt = 10
725  boussinesq = gv%Boussinesq
726  i_gearth = 1.0 / gv%g_Earth
727 
728  if (boussinesq) then
729  call mom_error(fatal,"Not yet converting thickness with Boussinesq approx.")
730  else
731  if (associated(tv%eqn_of_state)) then
732  do j=jsq,jeq+1 ; do i=isq,ieq+1
733  p_bot(i,j) = 0.0 ; p_top(i,j) = 0.0
734  enddo ; enddo
735  do k=1,nz
736  do j=js,je
737  do i=is,ie ; p_top(i,j) = p_bot(i,j) ; enddo
738  call calculate_density(tv%T(:,j,k), tv%S(:,j,k), p_top(:,j), rho, &
739  is, ie-is+1, tv%eqn_of_state)
740  do i=is,ie
741  p_bot(i,j) = p_top(i,j) + gv%g_Earth * h(i,j,k) * rho(i)
742  enddo
743  enddo
744 
745  do itt=1,max_itt
746  call int_specific_vol_dp(tv%T(:,:,k), tv%S(:,:,k), p_top, p_bot, &
747  0.0, g%HI, tv%eqn_of_state, dz_geo)
748  if (itt < max_itt) then ; do j=js,je
749  call calculate_density(tv%T(:,j,k), tv%S(:,j,k), p_bot(:,j), rho, &
750  is, ie-is+1, tv%eqn_of_state)
751  ! Use Newton's method to correct the bottom value.
752  ! The hydrostatic equation is linear to such a
753  ! high degree that no bounds-checking is needed.
754  do i=is,ie
755  p_bot(i,j) = p_bot(i,j) + rho(i) * (gv%g_Earth*h(i,j,k) - dz_geo(i,j))
756  enddo
757  enddo ; endif
758  enddo
759 
760  do j=js,je ; do i=is,ie
761  h(i,j,k) = (p_bot(i,j) - p_top(i,j)) * gv%kg_m2_to_H * i_gearth
762  enddo ; enddo
763  enddo
764  else
765  do k=1,nz ; do j=js,je ; do i=is,ie
766  h(i,j,k) = h(i,j,k) * gv%Rlay(k) * gv%kg_m2_to_H
767  enddo ; enddo ; enddo
768  endif
769  endif
770 
771 end subroutine convert_thickness

922 !> Adjust the layer thicknesses by cutting away the top at the depth where the hydrostatic
923 !! pressure matches p_surf
924 subroutine cut_off_column_top(nk, tv, Rho0, G_earth, depth, min_thickness, &
925  T, T_t, T_b, S, S_t, S_b, p_surf, h, remap_CS)
926  integer, intent(in) :: nk !< Number of layers
927  type(thermo_var_ptrs), intent(in) :: tv !< Thermodynamics structure
928  real, intent(in) :: Rho0 !< Reference density (kg/m3)
929  real, intent(in) :: G_earth !< Gravitational acceleration (m/s2)
930  real, intent(in) :: depth !< Depth of ocean column (m)
931  real, intent(in) :: min_thickness !< Smallest thickness allowed (m)
932  real, dimension(nk), intent(inout) :: T !< Layer mean temperature
933  real, dimension(nk), intent(in) :: T_t !< Temperature at top of layer
934  real, dimension(nk), intent(in) :: T_b !< Temperature at bottom of layer
935  real, dimension(nk), intent(inout) :: S !< Layer mean salinity
936  real, dimension(nk), intent(in) :: S_t !< Salinity at top of layer
937  real, dimension(nk), intent(in) :: S_b !< Salinity at bottom of layer
938  real, intent(in) :: p_surf !< Imposed pressure on ocean at surface (Pa)
939  real, dimension(nk), intent(inout) :: h !< Layer thickness (H units, m or Pa)
940  type(remapping_cs), pointer :: remap_CS ! Remapping structure for remapping T and S, if associated
941  ! Local variables
942  real, dimension(nk+1) :: e ! Top and bottom edge values for reconstructions
943  real, dimension(nk) :: h0, S0, T0, h1, S1, T1
944  real :: P_t, P_b, z_out, e_top
945  integer :: k
946 
947  ! Calculate original interface positions
948  e(nk+1) = -depth
949  do k=nk,1,-1
950  e(k) = e(k+1) + h(k)
951  h0(k) = h(nk+1-k) ! Keep a copy to use in remapping
952  enddo
953 
954  p_t = 0.
955  e_top = e(1)
956  do k=1,nk
957  call find_depth_of_pressure_in_cell(t_t(k), t_b(k), s_t(k), s_b(k), e(k), e(k+1), &
958  p_t, p_surf, rho0, g_earth, tv%eqn_of_state, p_b, z_out)
959  if (z_out>=e(k)) then
960  ! Imposed pressure was less that pressure at top of cell
961  exit
962  elseif (z_out<=e(k+1)) then
963  ! Imposed pressure was greater than pressure at bottom of cell
964  e_top = e(k+1)
965  else
966  ! Imposed pressure was fell between pressures at top and bottom of cell
967  e_top = z_out
968  exit
969  endif
970  p_t = p_b
971  enddo
972  if (e_top<e(1)) then
973  ! Clip layers from the top down, if at all
974  do k=1,nk
975  if (e(k)>e_top) then
976  ! Original e(K) is too high
977  e(k) = e_top
978  e_top = e_top - min_thickness ! Next interface must be at least this deep
979  endif
980  ! This layer needs trimming
981  h(k) = max( min_thickness, e(k) - e(k+1) )
982  if (e(k)<e_top) exit ! No need to go further
983  enddo
984  endif
985 
986  ! Now we need to remap but remapping assumes the surface is at the
987  ! same place in the two columns so we turn the column upside down.
988  if (associated(remap_cs)) then
989  do k=1,nk
990  s0(k) = s(nk+1-k)
991  t0(k) = t(nk+1-k)
992  h1(k) = h(nk+1-k)
993  enddo
994  call remapping_core_h(nk, h0, t0, nk, h1, t1, remap_cs )
995  call remapping_core_h(nk, h0, s0, nk, h1, s1, remap_cs )
996  do k=1,nk
997  s(k) = s1(nk+1-k)
998  t(k) = t1(nk+1-k)
999  enddo
1000  endif
1001 
1002 end subroutine cut_off_column_top
